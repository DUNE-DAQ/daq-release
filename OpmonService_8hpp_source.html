63std::shared_ptr<OpmonService>
64makeOpmonService(std::string const& service)
65{
66 TLOG() << "SERVICE = " << service;
67 auto sep = service.find("://");
68 std::string scheme;
69 if (sep == std::string::npos) { // simple path
70 scheme = "stdout";
71 } else { // with scheme
72 scheme = service.substr(0, sep);
73 }
74 std::string plugin_name = scheme + "OpmonService";
75 static cet::BasicPluginFactory bpf("duneOpmonService", "make");
76 std::shared_ptr<OpmonService> os_ptr;
77 try {
78 os_ptr = bpf.makePlugin<std::shared_ptr<OpmonService>>(plugin_name, service);
79 } catch (const cet::exception& cexpt) {
80 throw OpmonServiceCreationFailed(ERS_HERE, service, cexpt);
81 } catch (const ers::Issue& iexpt) {
82 throw OpmonServiceCreationFailed(ERS_HERE, service, iexpt);
83 } catch (...) { // NOLINT JCF Jan-27-2021 violates letter of the law but not the spirit
84 throw OpmonServiceCreationFailed(ERS_HERE, service, "Unknown error.");
85 }
86 return os_ptr;
87}
