215dfmessages::TriggerDecision
216TCProcessor::create_decision(const PendingTD& pending_td)
217{
218 m_earliest_tc_index = get_earliest_tc_index(pending_td);
219 TLOG_DEBUG(5) << "earliest TC index: " << m_earliest_tc_index;
220
221 if (pending_td.contributing_tcs.size() > 1) {
222 TLOG_DEBUG(5) << "!!! TD created from " << pending_td.contributing_tcs.size() << " TCs !!!";
223 }
224
225 dfmessages::TriggerDecision decision;
226 decision.trigger_number = 0; // filled by MLT
227 decision.run_number = 0; // filled by MLT
228 decision.trigger_timestamp = pending_td.contributing_tcs[m_earliest_tc_index].time_candidate;
229 decision.readout_type = dfmessages::ReadoutType::kLocalized;
230
231 if (m_hsi_passthrough == true) {
232 if (pending_td.contributing_tcs[m_earliest_tc_index].type == triggeralgs::TriggerCandidate::Type::kTiming) {
233 decision.trigger_type = pending_td.contributing_tcs[m_earliest_tc_index].detid & 0xff;
234 } else {
235 m_trigger_type_shifted = (static_cast<int>(pending_td.contributing_tcs[m_earliest_tc_index].type) << 8);
236 decision.trigger_type = m_trigger_type_shifted;
237 }
238 } else {
239 decision.trigger_type = 1; // m_trigger_type;
240 }
241
242 TLOG_DEBUG(3) << "HSI passthrough: " << m_hsi_passthrough
243 << ", TC detid: " << pending_td.contributing_tcs[m_earliest_tc_index].detid
244 << ", TC type: " << static_cast<int>(pending_td.contributing_tcs[m_earliest_tc_index].type)
245 << ", TC cont number: " << pending_td.contributing_tcs.size()
246 << ", DECISION trigger type: " << decision.trigger_type
247 << ", DECISION timestamp: " << decision.trigger_timestamp
248 << ", request window begin: " << pending_td.readout_start
249 << ", request window end: " << pending_td.readout_end;
250
251 std::vector<dfmessages::ComponentRequest> requests =
252 create_all_decision_requests(m_mandatory_links, pending_td.readout_start, pending_td.readout_end);
253 add_requests_to_decision(decision, requests);
254
255 if (!m_use_roi_readout) {
256 for (const auto& [key, value] : m_group_links) {
257 std::vector<dfmessages::ComponentRequest> group_requests =
258 create_all_decision_requests(value, pending_td.readout_start, pending_td.readout_end);
259 add_requests_to_decision(decision, group_requests);
260 }
261 } else { // using ROI readout
262 roi_readout_make_requests(decision);
263 }
264
265 return decision;
266}

284void
285TCProcessor::call_tc_decision(const TCProcessor::PendingTD& pending_td)
286{
287
288 if (m_use_bitwords) {
289 // Check trigger bitwords
290 m_TD_bitword = get_TD_bitword(pending_td);
291 m_bitword_check = check_trigger_bitwords();
292 }
293
294 if ((!m_use_bitwords) || (m_bitword_check)) {
295
296 dfmessages::TriggerDecision decision = create_decision(pending_td);
297 auto tn = decision.trigger_number;
298 auto td_ts = decision.trigger_timestamp;
299
300 if(!m_td_sink->try_send(std::move(decision), iomanager::Sender::s_no_block)) {
301 ers::warning(TDDropped(ERS_HERE, tn, td_ts));
302 m_tds_dropped++;
303 }
304 else {
305 m_new_tds++;
306 }
307 }

311void
312TCProcessor::add_tc(const triggeralgs::TriggerCandidate tc)
313{
314 bool added_to_existing = false;
315 int64_t tc_wallclock_arrived =
316 std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::steady_clock::now().time_since_epoch()).count();
317
318 if (m_tc_merging) {
319
320 for (std::vector<PendingTD>::iterator it = m_pending_tds.begin(); it != m_pending_tds.end();) {
321 if (check_overlap(tc, *it)) {
322 it->contributing_tcs.push_back(tc);
323 if ( (m_use_readout_map) && (m_readout_window_map.count(tc.type)) ){
324 TLOG_DEBUG(3) << "TC with start/end times " << tc.time_candidate - m_readout_window_map[tc.type].first << "/"
325 << tc.time_candidate + m_readout_window_map[tc.type].second
326 << " overlaps with pending TD with start/end times " << it->readout_start << "/"
327 << it->readout_end;
328 it->readout_start = ((tc.time_candidate - m_readout_window_map[tc.type].first) >= it->readout_start)
329 ? it->readout_start
330 : (tc.time_candidate - m_readout_window_map[tc.type].first);
331 it->readout_end = ((tc.time_candidate + m_readout_window_map[tc.type].second) >= it->readout_end)
332 ? (tc.time_candidate + m_readout_window_map[tc.type].second)
333 : it->readout_end;
334 } else {
335 TLOG_DEBUG(3) << "TC with start/end times " << tc.time_start << "/" << tc.time_end
336 << " overlaps with pending TD with start/end times " << it->readout_start << "/"
337 << it->readout_end;
338 it->readout_start = (tc.time_start >= it->readout_start) ? it->readout_start : tc.time_start;
339 it->readout_end = (tc.time_end >= it->readout_end) ? tc.time_end : it->readout_end;
340 }
341 it->walltime_expiration = tc_wallclock_arrived + m_buffer_timeout;
342 added_to_existing = true;
343 break;
344 }
345 ++it;
346 }
347 }
348
349 if (!added_to_existing) {
350 PendingTD td_candidate;
351 td_candidate.contributing_tcs.push_back(tc);
352 if ( (m_use_readout_map) && (m_readout_window_map.count(tc.type)) ){
353 td_candidate.readout_start = tc.time_candidate - m_readout_window_map[tc.type].first;
354 td_candidate.readout_end = tc.time_candidate + m_readout_window_map[tc.type].second;
355 } else {
356 td_candidate.readout_start = tc.time_start;
357 td_candidate.readout_end = tc.time_end;
358 }
359 td_candidate.walltime_expiration = tc_wallclock_arrived + m_buffer_timeout;
360 m_pending_tds.push_back(td_candidate);
361 }
362 return;
363}

365void
366TCProcessor::add_tc_ignored(const triggeralgs::TriggerCandidate tc)
367{
368 for (std::vector<PendingTD>::iterator it = m_pending_tds.begin(); it != m_pending_tds.end();) {
369 if (check_overlap(tc, *it)) {
370 if ( (m_use_readout_map) && (m_readout_window_map.count(tc.type)) ) {
371 TLOG_DEBUG(3) << "!Ignored! TC with start/end times " << tc.time_candidate - m_readout_window_map[tc.type].first
372 << "/" << tc.time_candidate + m_readout_window_map[tc.type].second
373 << " overlaps with pending TD with start/end times " << it->readout_start << "/"
374 << it->readout_end;
375 } else {
376 TLOG_DEBUG(3) << "!Ignored! TC with start/end times " << tc.time_start << "/" << tc.time_end
377 << " overlaps with pending TD with start/end times " << it->readout_start << "/"
378 << it->readout_end;
379 }
380 it->contributing_tcs.push_back(tc);
381 break;
382 }
383 ++it;
384 }
385 return;

388bool
389TCProcessor::check_overlap(const triggeralgs::TriggerCandidate& tc, const PendingTD& pending_td)
390{
391 if ( (m_use_readout_map) && (m_readout_window_map.count(tc.type)) ) {
392 return !(((tc.time_candidate + m_readout_window_map[tc.type].second) < pending_td.readout_start) ||
393 ((tc.time_candidate - m_readout_window_map[tc.type].first > pending_td.readout_end)));
394 } else {
395 return !((tc.time_end < pending_td.readout_start) || (tc.time_start > pending_td.readout_end));
396 }

399std::vector<TCProcessor::PendingTD>
400TCProcessor::get_ready_tds(std::vector<PendingTD>& pending_tds)
401{
402 std::vector<PendingTD> return_tds;
403 for (std::vector<PendingTD>::iterator it = pending_tds.begin(); it != pending_tds.end();) {
404 auto timestamp_now =
405 std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::steady_clock::now().time_since_epoch())
406 .count();
407 if (timestamp_now >= it->walltime_expiration) {
408 return_tds.push_back(*it);
409 it = pending_tds.erase(it);
410 } else if (check_td_readout_length(*it)) { // Also pass on TDs with (too) long readout window
411 return_tds.push_back(*it);
412 it = pending_tds.erase(it);
413 } else {
414 ++it;
415 }
416 }
417 return return_tds;
418}

470std::bitset<16>
471TCProcessor::get_TD_bitword(const PendingTD& ready_td)
472{
473 // get only unique types
474 std::vector<int> tc_types;
475 for (auto tc : ready_td.contributing_tcs) {
476 tc_types.push_back(static_cast<int>(tc.type));
477 }
478 tc_types.erase(std::unique(tc_types.begin(), tc_types.end()), tc_types.end());
479
480 // form TD bitword
481 std::bitset<16> td_bitword = 0b0000000000000000;
482 for (auto tc_type : tc_types) {
483 td_bitword.set(tc_type);
484 }
485 return td_bitword;

508bool
509TCProcessor::check_trigger_bitwords()
510{
511 bool trigger_check = false;
512 for (auto bitword : m_trigger_bitwords) {
513 TLOG_DEBUG(15) << "TD word: " << m_TD_bitword << ", bitword: " << bitword;
514 trigger_check = ((m_TD_bitword & bitword) == bitword);
515 TLOG_DEBUG(15) << "&: " << (m_TD_bitword & bitword);
516 TLOG_DEBUG(15) << "trigger?: " << trigger_check;
517 if (trigger_check == true)
518 break;
519 }
520 return trigger_check;

592dfmessages::ComponentRequest
593TCProcessor::create_request_for_link(dfmessages::SourceID link,
594 triggeralgs::timestamp_t start,
595 triggeralgs::timestamp_t end)
596{

691void
692TCProcessor::roi_readout_make_requests(dfmessages::TriggerDecision& decision)
693{
694 // Get configuration at random (weighted)
695 int group_pick = pick_roi_group_conf();
696 if (group_pick != -1) {
697 roi_group this_group = m_roi_conf[m_roi_conf_ids[group_pick]];
698 std::vector<dfmessages::SourceID> links;
699
700 // If mode is random, pick groups to request at random
701 if (this_group.mode == "kRandom") {
702 TLOG_DEBUG(10) << "RAND";
703 std::set<int> groups;
704 while (static_cast<int>(groups.size()) < this_group.n_links) {
705 groups.insert(get_random_num_int());
706 }
707 for (auto r_id : groups) {
708 links.insert(links.end(), m_group_links[r_id].begin(), m_group_links[r_id].end());
709 }
710 // Otherwise, read sequntially by IDs, starting at 0
711 } else {
712 TLOG_DEBUG(10) << "SEQ";
713 int r_id = 0;
714 while (r_id < this_group.n_links) {
715 links.insert(links.end(), m_group_links[r_id].begin(), m_group_links[r_id].end());
716 r_id++;
717 }
718 }
719
720 TLOG_DEBUG(10) << "TD timestamp: " << decision.trigger_timestamp;
721 TLOG_DEBUG(10) << "group window: " << this_group.time_window;
722
723 // Once the components are prepared, create requests and append them to decision
724 std::vector<dfmessages::ComponentRequest> requests =
725 create_all_decision_requests(links, decision.trigger_timestamp - this_group.time_window,
726 decision.trigger_timestamp + this_group.time_window);
727 add_requests_to_decision(decision, requests);
728 links.clear();
729 }
730 return;
731}
