114shell_expand_paths(const std::string& path)
115{
116
117 std::vector<std::string> paths;
118 wordexp_t substituted_path;
119 int code = wordexp(path.c_str(), &substituted_path, WRDE_NOCMD);
120 if (code)
121 throw runtime_error("Failed expanding path: " + path);
122
123 for (std::size_t i = 0; i != substituted_path.we_wordc; i++)
124 paths.push_back(substituted_path.we_wordv[i]);
125
126 wordfree(&substituted_path);
127
128 return paths;
129}

223double
224convert_bits_to_float(uint64_t bits, bool is_double_precision) // NOLINT(build/unsigned)
225{
226 uint32_t mantissa_shift = is_double_precision ? 52 : 23; // NOLINT(build/unsigned)
227 uint64_t exponent_mask = is_double_precision ? 0x7FF0000000000000 : 0x7f800000; // NOLINT(build/unsigned)
228 uint32_t bias = is_double_precision ? 1023 : 127; // NOLINT(build/unsigned)
229 uint32_t sign_shift = is_double_precision ? 63 : 31; // NOLINT(build/unsigned)
230
231 int32_t sign = (bits >> sign_shift) & 0x01;
232 uint32_t exponent_biased = ((bits & exponent_mask) >> mantissa_shift); // NOLINT(build/unsigned)
233 int32_t exponent = exponent_biased - bias;
234
235 int32_t power = -1;
236 double mantissa = 0.0;
237 for (uint32_t i = 0; i < mantissa_shift; ++i) { // NOLINT(build/unsigned)
238 uint64_t mantissa_bit = (bits >> (mantissa_shift - i - 1)) & 0x01; // NOLINT(build/unsigned)
239 mantissa += mantissa_bit * pow(2.0, power);
240 --power;
241 }
242
243 if (exponent_biased == 0) {
244 ++exponent;
245 if (mantissa == 0)
246 return 0;
247 } else {
248 mantissa += 1.0;
249 }
250 return (sign ? -1 : 1) * pow(2.0, exponent) * mantissa;
251}

300locate(float xx[], unsigned long n, float x) // NOLINT
301{
302 uint32_t j, ju, jm, jl; // NOLINT(build/unsigned)
303 int ascnd;
304 jl = 0; // Initialize lower
305 ju = n + 1; // and upper limits.
306 ascnd = (xx[n] >= xx[1]); // NOLINT
307
308 while (ju - jl > 1) // If we are not yet done,
309 {
310 jm = (ju + jl) >> 1; // compute a midpoint,
311
312 if ((x >= xx[jm]) == ascnd) // added additional parenthesis
313 {
314 jl = jm; // and replace either the lower limit
315 } else {
316 ju = jm; // or the upper limit, as appropriate.
317 }
318 } // Repeat until the test condition is satisﬁed.
319
320 if (x == xx[1]) { // NOLINT
321 j = 1; // Then set the output
322 } else if (x == xx[n]) { // NOLINT
323 j = n - 1;
324 } else {
325 j = jl;
326 }
327
328 return j;
329}
