82void
83TimingHardwareManager::scrap()
84{
85 m_hw_command_receiver->remove_callback();
86
87 auto time_of_scrap = std::chrono::high_resolution_clock::now();
88 while(m_command_threads.size())
89 {
90 auto now = std::chrono::high_resolution_clock::now();
91 auto ms_since_scrap = std::chrono::duration_cast<std::chrono::milliseconds>(now - time_of_scrap);
92 TLOG_DEBUG(0) << "Have been waiting for " << ms_since_scrap.count() << " ms for " << m_command_threads.size() << " command threads to finish...";
93 std::this_thread::sleep_for(std::chrono::microseconds(250000));
94 }
95 m_run_endpoint_scan_cleanup_thread.store(false);
96
97 stop_hw_mon_gathering();
98
99 scrap_uhal();
100
101 m_command_threads.clear();
102 m_info_gatherers.clear();
103 m_timing_hw_cmd_map_.clear();
104 m_hw_device_map.clear();
105 m_connection_manager.reset();
106}

108const timing::TimingNode*
109TimingHardwareManager::get_timing_device_plain(const std::string& device_name)
110{
111
112 if (!device_name.compare("")) {
113 std::stringstream message;
114 message << "UHAL device name is an empty string";
115 throw UHALDeviceNameIssue(ERS_HERE, message.str());
116 }
117
118 if (auto hw_device_entry = m_hw_device_map.find(device_name); hw_device_entry != m_hw_device_map.end()) {
119 return dynamic_cast<const timing::TimingNode*>(&hw_device_entry->second->getNode(""));
120 } else {
121 TLOG_DEBUG(0) << get_name() << ": hw device interface for: " << device_name

123

129 std::stringstream message;

131 throw UHALDeviceNameIssue(ERS_HERE, message.str(), exception);

133
134 TLOG_DEBUG(0) << get_name() << ": hw device interface for: " << device_name << " successfully created.";

140void
141TimingHardwareManager::gather_monitor_data(InfoGatherer& gatherer)
142{
143 auto device_name = gatherer.get_device_name();
144
145 while (gatherer.run_gathering()) {
146
147 // collect the data from the hardware
148 try {
149 auto design = get_timing_device_plain(device_name);
150 gatherer.collect_info_from_device(*design);
151 } catch (const std::exception& excpt) {
152 ers::warning(FailedToCollectOpMonInfo(ERS_HERE, device_name, excpt));
153 }
154
155 auto prev_gather_time = std::chrono::steady_clock::now();
156 auto next_gather_time = prev_gather_time + std::chrono::microseconds(gatherer.get_gather_interval());
157
158 // check running_flag periodically
159 auto slice_period = std::chrono::microseconds(10000);
160 auto next_slice_gather_time = prev_gather_time + slice_period;
161
162 bool break_flag = false;
163 while (next_gather_time > next_slice_gather_time + slice_period) {
164 if (!gatherer.run_gathering()) {
165 TLOG_DEBUG(0) << "while waiting to gather data, negative run gatherer flag detected.";
166 break_flag = true;
167 break;
168 }
169 std::this_thread::sleep_until(next_slice_gather_time);
170 next_slice_gather_time = next_slice_gather_time + slice_period;
171 }
172 if (break_flag == false) {
173 std::this_thread::sleep_until(next_gather_time);
174 }
175 }

209 it->second.get()->start_gathering_thread();

211 }
212 if (!gatherer_found) ers::warning(AttemptedToControlNonExantInfoGatherer(ERS_HERE, "start", device_name));

253void
254TimingHardwareManager::process_hardware_command(timingcmd::TimingHwCmd& timing_hw_cmd)
255{
256 std::ostringstream starting_stream;
257 starting_stream << ": Executing process_hardware_command() callback.";
258 TLOG_DEBUG(0) << get_name() << starting_stream.str();
259
260 ++m_received_hw_commands_counter;
261
262 TLOG_DEBUG(0) << get_name() << ": Received hardware command #" << m_received_hw_commands_counter.load()
263 << ", it is of type: " << timing_hw_cmd.id << ", targeting device: " << timing_hw_cmd.device;
264
265 std::string hw_cmd_name = timing_hw_cmd.id;
266 if (auto cmd = m_timing_hw_cmd_map_.find(hw_cmd_name); cmd != m_timing_hw_cmd_map_.end()) {
267
268 ++m_accepted_hw_commands_counter;
269
270 TLOG_DEBUG(0) << "Found hw cmd: " << hw_cmd_name;
271 try {
272 std::invoke(cmd->second, timing_hw_cmd);
273 } catch (const std::exception& exception) {
274 ers::error(FailedToExecuteHardwareCommand(ERS_HERE, hw_cmd_name, timing_hw_cmd.device, exception));
275 ++m_failed_hw_commands_counter;
276 }
277 } else {
278 ers::error(InvalidHardwareCommandID(ERS_HERE, hw_cmd_name));
279 ++m_rejected_hw_commands_counter;
280 }
281
282 std::ostringstream exiting_stream;
283 exiting_stream << ": Finished executing process_hardware_command() callback. Received " << m_received_hw_commands_counter.load()
284 << " commands";
285 TLOG_DEBUG(0) << get_name() << exiting_stream.str();
286}

289void
290TimingHardwareManager::io_reset(const timingcmd::TimingHwCmd& hw_cmd)
291{
292 timingcmd::IOResetCmdPayload cmd_payload;
293 timingcmd::from_json(hw_cmd.payload, cmd_payload);
294
295 TLOG_DEBUG(0) << get_name() << ": " << hw_cmd.device << " io reset";
296
297 // io reset disrupts hw mon gathering, so stop if running
298 auto running_hw_gatherers = check_hw_mon_gatherer_is_running(hw_cmd.device);
299 for (auto& gatherer: running_hw_gatherers)
300 {
301 stop_hw_mon_gathering(gatherer);
302 }
303
304 auto design = get_timing_device<const timing::TopDesignInterface*>(hw_cmd.device);
305
306 if (cmd_payload.soft) {
307 TLOG_DEBUG(0) << get_name() << ": " << hw_cmd.device << " soft io reset";
308 design->soft_reset_io();
309 } else {
310 TLOG_DEBUG(0) << get_name() << ": " << hw_cmd.device
311 << " io reset, with supplied clk file: " << cmd_payload.clock_config << "and fanout mode: " << cmd_payload.fanout_mode;
312 design->reset_io(cmd_payload.fanout_mode, cmd_payload.clock_config);
313 }
314
315 // if hw mon gathering was running previously, start it again
316 for (auto& gatherer: running_hw_gatherers)
317 {
318 start_hw_mon_gathering(gatherer);
319 }
320}

341void
342TimingHardwareManager::master_endpoint_scan(const timingcmd::TimingHwCmd& hw_cmd)
343{
344 TLOG_DEBUG(0) << get_name() << ": " << hw_cmd.device << " master_endpoint_scan";
345
346 std::stringstream command_thread_uid;
347 auto t = std::time(nullptr);
348 auto tm = *std::localtime(&t);
349 command_thread_uid << "enpoint_scan_cmd_at_" << std::put_time(&tm, "%d-%m-%Y %H-%M-%S") << "_cmd_num_" << m_accepted_hw_commands_counter.load();
350
351 if (m_command_threads.size() > 5)
352 {
353 ers::warning(TooManyEndpointScanThreadsQueued(ERS_HERE, m_command_threads.size()));
354 }
355 else
356 {
357 TLOG_DEBUG(1) << "Queuing: " << command_thread_uid.str();
358
359 auto thread_key = command_thread_uid.str();
360 std::unique_lock map_lock(m_command_threads_map_mutex);
361
362 m_command_threads.emplace(thread_key, std::make_unique<std::thread>(std::bind(&TimingHardwareManager::perform_endpoint_scan, this, hw_cmd)));
363 }

366void TimingHardwareManager::perform_endpoint_scan(const timingcmd::TimingHwCmd& hw_cmd)
367{
368 timingcmd::TimingMasterEndpointScanPayload cmd_payload;
369 timingcmd::from_json(hw_cmd.payload, cmd_payload);
370
371 for (auto& endpoint_location : cmd_payload.endpoints)
372 {
373 auto endpoint_address = endpoint_location.address;
374 auto fanout_slot = endpoint_location.fanout_slot;
375 auto sfp_slot = endpoint_location.sfp_slot;
376
377 std::unique_lock<std::mutex> master_sfp_lock(master_sfp_mutex);
378
379 TLOG_DEBUG(1) << get_name() << ": " << hw_cmd.device << " master_endpoint_scan starting: ept adr: " << endpoint_address << ", ept sfp: " << sfp_slot;
380
381 auto master_design = get_timing_device<const timing::MasterDesignInterface*>(hw_cmd.device);
382
383 try
384 {
385 master_design->get_master_node_plain()->switch_endpoint_sfp(endpoint_address, true);
386
387 if (sfp_slot >= 0)
388 {
389 if (fanout_slot > 0)
390 {
391 // configure fanout receiver
392 get_timing_device<const timing::FanoutDesign*>(m_monitored_device_names_fanout.at(fanout_slot-1))->switch_downstream_mux_channel(sfp_slot, false);
393 }
394 else
395 {
396 dynamic_cast<const timing::MasterMuxDesign*>(master_design)->switch_downstream_mux_channel(sfp_slot, false);
397 }
398 }
399 // configure any master mux, possibly
400 master_design->get_master_node_plain()->scan_endpoint(endpoint_address, false);
401 master_design->get_master_node_plain()->switch_endpoint_sfp(endpoint_address, false);
402 }
403 catch(std::exception& e)
404 {
405 ers::error(EndpointScanFailure(ERS_HERE,e));
406 master_design->get_master_node_plain()->switch_endpoint_sfp(endpoint_address, false);
407 }
408 }

411void TimingHardwareManager::clean_endpoint_scan_threads()
412{
413 TLOG_DEBUG(0) << "Entering clean_endpoint_scan_threads()";
414 bool break_flag = false;
415 while (!break_flag)
416 {
417 for (auto& thread : m_command_threads)
418 {
419 if (thread.second->joinable())
420 {
421 std::unique_lock map_lock(m_command_threads_map_mutex);
422 TLOG_DEBUG(2) << thread.first << " thread ready. Cleaning up.";
423 thread.second->join();
424 m_command_threads.erase(thread.first);
425 }
426 }
427
428 auto prev_clean_time = std::chrono::steady_clock::now();
429 auto next_clean_time = prev_clean_time + std::chrono::milliseconds(30);
430
431 // check running_flag periodically
432 auto flag_check_period = std::chrono::milliseconds(1);
433 auto next_flag_check_time = prev_clean_time + flag_check_period;
434
435 while (next_clean_time > next_flag_check_time + flag_check_period) {
436 if (!m_run_endpoint_scan_cleanup_thread.load()) {
437 TLOG_DEBUG(2) << "while waiting to clean up endpoint scan threads, negative run gatherer flag detected.";
438 break_flag = true;
439 break;
440 }
441 std::this_thread::sleep_until(next_flag_check_time);
442 next_flag_check_time = next_flag_check_time + flag_check_period;
443 }
444 if (break_flag == false) {
445 std::this_thread::sleep_until(next_clean_time);
446 }
447 }
448 TLOG_DEBUG(0) << "Exiting clean_endpoint_scan_threads()";
449}
