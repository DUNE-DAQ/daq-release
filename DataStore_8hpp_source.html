147inline std::unique_ptr<DataStore>
148make_data_store(const std::string& type, const nlohmann::json& conf)
149{
150 static cet::BasicPluginFactory bpf("duneDataStore", "make"); // NOLINT
151
152 std::unique_ptr<DataStore> ds;
153 try {
154 ds = bpf.makePlugin<std::unique_ptr<DataStore>>(type, conf);
155 } catch (const cet::exception& cexpt) {
156 throw DataStoreCreationFailed(ERS_HERE, type, conf, cexpt);
157 }
158
159 return ds;
160}
