91 const std::vector<std::string> construct_commandline_parameters_appfwk(const T* app,
92 const conffwk::Configuration& confdb,
93 const dunedaq::coredal::Session* session) {
94
95 const dunedaq::coredal::Service* control_service = nullptr;
96
97 for (auto const* as: app->get_exposes_service())
98 if (as->UID() == app->UID()+"_control") // unclear this is the best way to do this.
99 control_service = as;
100
101 if (control_service == nullptr)
102 throw NoControlServiceDefined(ERS_HERE, app->UID());
103
104 const std::string control_uri =
105 control_service->get_protocol()
106 + "://"
107 + app->get_runs_on()->get_runs_on()->UID()
108 + ":"
109 + std::to_string(control_service->get_port());
110
111 const dunedaq::coredal::Application* opmon_app = nullptr;
112 for (auto const* ia: session->get_infrastructure_applications())
113 if (ia->castable("OpMonService"))
114 opmon_app = ia;
115
116 if (opmon_app == nullptr)
117 throw NoOpmonInfrastructure(ERS_HERE, session->UID());
118
119 const dunedaq::coredal::Service* opmon_service = opmon_app->get_exposes_service()[0];
120 std::string opmon_uri =
121 opmon_service->get_protocol()
122 + "://"
123 + opmon_app->get_runs_on()->get_runs_on()->UID()
124 + ":"
125 + std::to_string(opmon_service->get_port())
126 + opmon_service->get_path();
127
128 if (opmon_service->get_protocol() == "file")
129 opmon_uri =
130 opmon_service->get_protocol()
131 + "://"
132 + opmon_service->get_path();
133
134 const std::string configuration_uri = confdb.get_impl_spec();
135
136 return {
137 "--name",
138 app->UID(),
139 "-c",
140 control_uri,
141 "-i",
142 opmon_uri,
143 "--configurationService",
144 configuration_uri,
145 };
146 }
