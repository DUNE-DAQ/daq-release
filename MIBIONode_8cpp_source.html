77MIBIONode::reset(int32_t fanout_mode, const std::string& clock_config_file) const
78{
79
80 write_soft_reset_register();
81
82 millisleep(1000);
83
84 // Find the right pll config file
85 std::string clock_config_path = get_full_clock_config_file_path(clock_config_file, fanout_mode);
86 TLOG() << "PLL configuration file : " << clock_config_path;
87
88 // Upload config file to PLL
89 configure_pll(clock_config_path);
90
91 // Reset mmcm
92 getNode("csr.ctrl.rst").write(0x1);
93 getNode("csr.ctrl.rst").write(0x0);
94
95 getNode("io_select.csr.ctrl.amc_out").write(0xfff);
96 getNode("io_select.csr.ctrl.amc_in").write(0x0);
97 getNode("io_select.csr.ctrl.usfp_src").write(0x0);
98
99 getClient().dispatch();
100
101 TLOG() << "Reset done";
102}

164

167 i2c_switch->set_channels_states(1UL << sfp_id);

169 auto sfp = get_i2c_device<I2CSFPSlave>(m_sfp_i2c_buses.at(0), "SFP_EEProm");
170
171 status << "SFP " << sfp_id << ":" << std::endl;
172

175 status << sfp->get_status();

177 catch(...)

179 i2c_switch->set_channels_states(8);

182
183 i2c_switch->set_channels_states(8);
184
185 if (print_out)
186 TLOG() << status.str();
187
188 return status.str();

232void
233MIBIONode::get_info(opmonlib::InfoCollector& ci, int level) const
234{
235 auto i2c_switch = get_i2c_device<I2C9546SwitchSlave>("i2c", "TCA9546_Switch");
236
237 if (level >= 2) {
238 i2c_switch->set_channels_states(8);
239
240 timinghardwareinfo::TimingPLLMonitorData pll_mon_data;
241 get_pll()->get_info(pll_mon_data);
242 ci.add(pll_mon_data);
243
244 for (uint i=0; i < 3; ++i)
245 {
246 opmonlib::InfoCollector sfp_ic;
247
248 // enable i2c path for sfp
249 i2c_switch->set_channels_states(1UL << i);
250
251 auto sfp = this->get_i2c_device<I2CSFPSlave>(m_sfp_i2c_buses.at(0), "SFP_EEProm");
252
253 try
254 {
255 sfp->get_info(sfp_ic, level);
256 }
257 catch (timing::SFPUnreachable& e)
258 {
259 // It is valid that an SFP may not be installed, currently no good way of knowing whether they it should be
260 TLOG_DEBUG(2) << "Failed to communicate with SFP " << i << " on I2C switch channel " << (1UL << i) << " on i2c bus" << m_sfp_i2c_buses.at(0);
261 continue;
262 }
263 ci.add("sfp_"+std::to_string(i),sfp_ic);
264 }
265 i2c_switch->set_channels_states(8);
266 }
267 if (level >= 1) {
268 timinghardwareinfo::TimingMIBMonitorData mon_data;
269 this->get_info(mon_data);
270 ci.add(mon_data);
271 }
272}
