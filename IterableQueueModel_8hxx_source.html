6// Free allocated memory that is different for alignment strategies and allocation policies
7template<class T>
8void
9IterableQueueModel<T>::free_memory()
10{
11 // We need to destruct anything that may still exist in our queue.
12 // (No real synchronization needed at destructor time: only one
13 // thread can be doing this.)
14 if (!std::is_trivially_destructible<T>::value) {
15 std::size_t readIndex = readIndex_;
16 std::size_t endIndex = writeIndex_;
17 while (readIndex != endIndex) {
18 records_[readIndex].~T();
19 if (++readIndex == size_) { // NOLINT(runtime/increment_decrement)
20 readIndex = 0;
21 }
22 }
23 }
24 // Different allocators require custom free functions
25 if (intrinsic_allocator_) {
26 _mm_free(records_);
27 } else if (numa_aware_) {
28#ifdef WITH_LIBNUMA_SUPPORT
29 numa_free(records_, sizeof(T) * size_);
30#endif
31 } else {
32 std::free(records_);
33 }
34}

36// Allocate memory based on different alignment strategies and allocation policies
37template<class T>
38void
39IterableQueueModel<T>::allocate_memory(std::size_t size,
40 bool numa_aware,
41 uint8_t numa_node, // NOLINT (build/unsigned)
42 bool intrinsic_allocator,
43 std::size_t alignment_size)
44{
45 assert(size >= 2);
46 // TODO: check for valid alignment sizes! | July-21-2021 | Roland Sipos | rsipos@cern.ch
47
48 if (intrinsic_allocator && alignment_size > 0) { // _mm allocator
49 records_ = static_cast<T*>(_mm_malloc(sizeof(T) * size, alignment_size));
50
51 } else if (!intrinsic_allocator && alignment_size > 0) { // std aligned allocator
52 records_ = static_cast<T*>(std::aligned_alloc(alignment_size, sizeof(T) * size));
53
54 } else if (numa_aware && numa_node < 8) { // numa allocator from libnuma; we get "numa_node >= 0" for free, given its datatype
55#ifdef WITH_LIBNUMA_SUPPORT
56 numa_set_preferred((unsigned)numa_node); // https://linux.die.net/man/3/numa_set_preferred
57 #ifdef WITH_LIBNUMA_BIND_POLICY
58 numa_set_bind_policy(WITH_LIBNUMA_BIND_POLICY); // https://linux.die.net/man/3/numa_set_bind_policy
59 #endif
60 #ifdef WITH_LIBNUMA_STRICT_POLICY
61 numa_set_strict(WITH_LIBNUMA_STRICT_POLICY); // https://linux.die.net/man/3/numa_set_strict
62 #endif
63 records_ = static_cast<T*>(numa_alloc_onnode(sizeof(T) * size, numa_node));
64#else
65 throw GenericConfigurationError(ERS_HERE,
66 "NUMA allocation was requested but program was built without USE_LIBNUMA");
67#endif
68 } else if (!numa_aware && !intrinsic_allocator && alignment_size == 0) {
69 // Standard allocator
70 records_ = static_cast<T*>(std::malloc(sizeof(T) * size));
71
72 } else {
73 // Let it fail, as expected combination might be invalid
74 // records_ = static_cast<T*>(std::malloc(sizeof(T) * size_);
75 }
76
77 size_ = size;
78 numa_aware_ = numa_aware;
79 numa_node_ = numa_node;
80 intrinsic_allocator_ = intrinsic_allocator;
81 alignment_size_ = alignment_size;
82}

84template<class T>
85void
86IterableQueueModel<T>::prefill_task()
87{
88 // Wait until LB issues ready
89 std::unique_lock lk(prefill_mutex_);
90 prefill_cv_.wait(lk, [this]{ return prefill_ready_; });
91
92 // After wait, we are ready to force page-fault
93 for (size_t i = 0; i < size_ - 1; ++i) {
94 T element = T();
95 write_(std::move(element));
96 }
97 flush();
98
99 // Preallocation done
100 prefill_done_ = true;
101
102 // Manual unlock is done before notify: avoid waking up the waiting thread only to block again.
103 lk.unlock();
104 prefill_cv_.notify_one();
105}

107template<class T>
108void
109IterableQueueModel<T>::force_pagefault()
110{
111 // Local prefiller thread
112 std::thread prefill_thread(&IterableQueueModel<T>::prefill_task, this);
113
114 // Tweak prefiller thread
115 char tname[16];
116 snprintf(tname, 16, "%s-%d", prefiller_name_.c_str(), numa_node_);
117 auto handle = prefill_thread.native_handle();
118 pthread_setname_np(handle, tname);
119
120#ifdef WITH_LIBNUMA_SUPPORT
121 cpu_set_t affinitymask;
122 CPU_ZERO(&affinitymask);
123 struct bitmask *nodecpumask = numa_allocate_cpumask();
124 int ret = 0;
125 // Get NODE CPU mask
126 ret = numa_node_to_cpus(numa_node_, nodecpumask);
127 assert(ret == 0);
128 // Apply corresponding NODE CPUs to affinity mask
129 for (int i=0; i< numa_num_configured_cpus(); ++i) {
130 if (numa_bitmask_isbitset(nodecpumask, i)) {
131 CPU_SET(i, &affinitymask);
132 }
133 }
134 ret = pthread_setaffinity_np(handle, sizeof(cpu_set_t), &affinitymask);
135 assert(ret == 0);
136 numa_free_cpumask(nodecpumask);
137#endif
138
139 // Trigger prefiller thread
140 {
141 std::lock_guard lk(prefill_mutex_);
142 prefill_ready_ = true;
143 }
144 prefill_cv_.notify_one();
145 // Wait for prefiller thread to finish
146 {
147 std::unique_lock lk(prefill_mutex_);
148 prefill_cv_.wait(lk, [this]{ return prefill_done_; });
149 }
150 // Join with prefiller thread
151 prefill_thread.join();
152}

154// Write element into the queue
155template<class T>
156bool
157IterableQueueModel<T>::write(T&& record)
158{
159 auto const currentWrite = writeIndex_.load(std::memory_order_relaxed);
160 auto nextRecord = currentWrite + 1;
161 if (nextRecord == size_) {
162 nextRecord = 0;
163 }
164
165 if (nextRecord != readIndex_.load(std::memory_order_acquire)) {
166 new (&records_[currentWrite]) T(std::move(record));
167 writeIndex_.store(nextRecord, std::memory_order_release);
168 return true;
169 }
170
171 // queue is full
172 ++overflow_ctr;
173 return false;
174}

176// Read element from a queue (move or copy the value at the front of the queue to given variable)
177template<class T>
178bool
179IterableQueueModel<T>::read(T& record)
180{
181 auto const currentRead = readIndex_.load(std::memory_order_relaxed);
182 if (currentRead == writeIndex_.load(std::memory_order_acquire)) {
183 // queue is empty
184 return false;
185 }
186
187 auto nextRecord = currentRead + 1;
188 if (nextRecord == size_) {
189 nextRecord = 0;
190 }
191 record = std::move(records_[currentRead]);
192 records_[currentRead].~T();
193 readIndex_.store(nextRecord, std::memory_order_release);
194 return true;
195}

232// Returns true if write index reached read index
233template<class T>
234bool
235IterableQueueModel<T>::isFull() const
236{
237 auto nextRecord = writeIndex_.load(std::memory_order_acquire) + 1;
238 if (nextRecord == size_) {
239 nextRecord = 0;
240 }
241 if (nextRecord != readIndex_.load(std::memory_order_acquire)) {
242 return false;
243 }
244 // queue is full
245 return true;
246}

248// Returns a good-enough guess on current occupancy:
249// * If called by consumer, then true size may be more (because producer may
250// be adding items concurrently).
251// * If called by producer, then true size may be less (because consumer may
252// be removing items concurrently).
253// * It is undefined to call this from any other thread.
254template<class T>
255std::size_t
256IterableQueueModel<T>::occupancy() const
257{
258 int ret = static_cast<int>(writeIndex_.load(std::memory_order_acquire)) -
259 static_cast<int>(readIndex_.load(std::memory_order_acquire));
260 if (ret < 0) {
261 ret += static_cast<int>(size_);
262 }
263 return static_cast<std::size_t>(ret);
264}

266// Gives a pointer to the current read index
267template<class T>
268const T*
269IterableQueueModel<T>::front()
270{
271 auto const currentRead = readIndex_.load(std::memory_order_relaxed);
272 if (currentRead == writeIndex_.load(std::memory_order_acquire)) {
273 return nullptr;
274 }
275 return &records_[currentRead];
276}

278// Gives a pointer to the current write index
279template<class T>
280const T*
281IterableQueueModel<T>::back()
282{
283 auto const currentWrite = writeIndex_.load(std::memory_order_relaxed);
284 if (currentWrite == readIndex_.load(std::memory_order_acquire)) {
285 return nullptr;
286 }
287 int currentLast = currentWrite;
288 if (currentLast == 0) {
289 currentLast = size_ - 1;
290 } else {
291 currentLast--;
292 }
293 return &records_[currentLast];
294}
