31resolve_environment_variables(std::string& input_string)
32{
33 static std::regex env_var_pattern("\\$\\{([^}]+)\\}");
34 std::smatch match;
35 while (std::regex_search(input_string, match, env_var_pattern)) {
36 const char* s = getenv(match[1].str().c_str());

38 input_string.replace(match[0].first, match[0].second, env_var);
