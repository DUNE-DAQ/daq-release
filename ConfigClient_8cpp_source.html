27ConfigClient::ConfigClient(const std::string& server,
28 const std::string& port,
29 std::chrono::milliseconds publish_interval)
30{
31 char* session = getenv("DUNEDAQ_SESSION");
32 if (session) {
33 m_session = std::string(session);
34 } else {
35 session = getenv("DUNEDAQ_PARTITION");
36 if (session) {
37 m_session = std::string(session);
38 } else {
39 throw(EnvNotFound(ERS_HERE, "DUNEDAQ_SESSION"));
40 }
41 }
42
43 tcp::resolver resolver(m_ioContext);
44 m_addr = resolver.resolve(server, port);
45 m_active = true;
46 m_thread = std::thread([this, publish_interval]() {
47 while (m_active) {
48 try {
49 publish();
50 m_connected = true;
51 TLOG_DEBUG(24) << "Automatic publish complete";
52 } catch (ers::Issue& ex) {
53 if (m_connected)
54 ers::error(ex);
55 else
56 TLOG() << ex;
57 } catch (std::exception& ex) {
58 auto ers_ex = PublishException(ERS_HERE, ex.what());
59 if (m_connected)
60 ers::error(ers_ex);
61 else
62 TLOG() << ers_ex;
63 }
64 std::this_thread::sleep_for(publish_interval);
65 }
66 retract();
67 if (!m_connected) {
68 ers::error(PublishException(ERS_HERE, "Publish thread was unable to publish to Connectivity Service!"));
69 }
70 });
71}
