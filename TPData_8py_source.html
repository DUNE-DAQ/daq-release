63 def load_frag(self, frag_path) -> np.ndarray:
64 """
65 Load a fragment from a given fragment path
66 and append to self.tp_data.
67
68 Returns an np.ndarray of dtype self.tp_dt.
69 """
70 frag = self._h5_file.get_frag(frag_path)
71
72 data_size = frag.get_data_size()
73 tp_size = trgdataformats.TriggerPrimitive.sizeof()
74 num_tps = data_size // tp_size
75 if (not self._quiet):
76 print(f"Loaded frag with {num_tps} TPs.")
77
78 np_tp_data = np.zeros((num_tps,), dtype=self.tp_dt)
79 for idx, byte_idx in enumerate(range(0, data_size, tp_size)): # Increment by TP size
80 tp_datum = trgdataformats.TriggerPrimitive(frag.get_data(byte_idx))
81
82 np_tp_data[idx] = np.array([(
83 tp_datum.adc_integral,
84 tp_datum.adc_peak,
85 tp_datum.algorithm,
86 tp_datum.channel,
87 tp_datum.detid,
88 tp_datum.flag,
89 tp_datum.time_over_threshold,
90 tp_datum.time_peak,
91 tp_datum.time_start,
92 tp_datum.type,
93 tp_datum.version)],
94 dtype=self.tp_dt)
95 self.tp_data.append(np_tp_data)
96
97 return np_tp_data
