107void dbse::SchemaGraphicsScene::dropEvent ( QGraphicsSceneDragDropEvent * event )
108{
109 QByteArray encodedData = event->mimeData()->data ( "application/vnd.text.list" );
110 QDataStream stream ( &encodedData, QIODevice::ReadOnly );
111 QStringList SchemaClasses;
112
113 while ( !stream.atEnd() )
114 {
115 QString ClassName;
116 stream >> ClassName;
117 SchemaClasses.append ( ClassName );
118 }
119
120 QList<QPointF> Positions;
121
122 for ( int i = 0; i < SchemaClasses.size(); ++i )
123 {
124 Positions.push_back ( event->scenePos() );
125 }
126
127 AddItemToScene ( SchemaClasses, Positions );
128}

206void dbse::SchemaGraphicsScene::AddItemToScene ( QStringList SchemaClasses,
207 QList<QPointF> Positions )
208{
209 for ( QString & ClassName : SchemaClasses )
210 {
211 if ( !ItemMap.contains ( ClassName ) )
212 {
213 SchemaGraphicObject * Object = new SchemaGraphicObject ( ClassName );
214 Object->setPos ( Positions.at ( SchemaClasses.indexOf ( ClassName ) ) );
215 Object->set_indirects_visibility(m_indirects_visible);
216 addItem ( Object );
218 ItemMap.insert ( ClassName, Object );
219 }
220 }
221
222 for ( QString & ClassName : ItemMap.keys() )
223 {
224 OksClass * ClassInfo = KernelWrapper::GetInstance().FindClass ( ClassName.toStdString() );
225
226 const std::list<OksRelationship *> * DirectRelationshipList =
227 ClassInfo->direct_relationships();
228 const std::list<std::string *> * DirectSuperClassesList = ClassInfo->direct_super_classes();
229
231 if ( DirectRelationshipList != nullptr )
232 {
233 for ( OksRelationship * ClassRelationship : * ( DirectRelationshipList ) )
234 {
235 QString RelationshipClassType = QString::fromStdString (
236 ClassRelationship->get_class_type()->get_name() );
237
238 if ( ItemMap.contains ( RelationshipClassType ) && !ItemMap[ClassName]->HasArrow (
239 ItemMap[RelationshipClassType] ) )
240 {
241 QString SchemaCardinality =
242 KernelWrapper::GetInstance().GetCardinalityStringRelationship ( ClassRelationship );
243 SchemaGraphicArrow * NewArrow = new SchemaGraphicArrow (
244 ItemMap[ClassName], ItemMap[RelationshipClassType], false,
245 ClassRelationship->get_is_composite(),
246 QString::fromStdString ( ClassRelationship->get_name() ), SchemaCardinality );
247 ItemMap[ClassName]->AddArrow ( NewArrow );
248 ItemMap[RelationshipClassType]->AddArrow ( NewArrow );
249 addItem ( NewArrow );
250 //NewArrow->SetLabelScene(this);
251 NewArrow->setZValue ( -1000.0 );
252 NewArrow->UpdatePosition();
253 }
254 }
255 }
256
258 if ( DirectSuperClassesList != nullptr )
259 {
260 for ( std::string * SuperClassNameStd : * ( DirectSuperClassesList ) )
261 {
262 QString SuperClassName = QString::fromStdString ( *SuperClassNameStd );
263
264 if ( ItemMap.contains ( SuperClassName ) && !ItemMap[ClassName]->HasArrow (
265 ItemMap[SuperClassName] ) )
266 {
267 SchemaGraphicArrow * NewArrow = new SchemaGraphicArrow ( ItemMap[ClassName],
268 ItemMap[SuperClassName], true,
269 false, "", "" );
270 ItemMap[ClassName]->AddArrow ( NewArrow );
271 ItemMap[SuperClassName]->AddArrow ( NewArrow );
272 addItem ( NewArrow );
273 //NewArrow->SetLabelScene(this);
274 NewArrow->setZValue ( -1000.0 );
275 NewArrow->UpdatePosition();
276 }
277 }
278 }

330void dbse::SchemaGraphicsScene::mouseReleaseEvent ( QGraphicsSceneMouseEvent * mouseEvent )
331{
332 if ( line != nullptr )
333 {
334 QList<QGraphicsItem *> startItems = items ( line->line().p1() );
335
336 if ( startItems.count() && startItems.first() == line )
337 {
338 startItems.removeFirst();
339 }
340
341 QList<QGraphicsItem *> endItems = items ( line->line().p2() );
342
343 if ( endItems.count() && endItems.first() == line )
344 {
345 endItems.removeFirst();
346 }
347
348 removeItem ( line );
349 delete line;
350
351 if ( startItems.count() > 0 && endItems.count() > 0
352 && startItems.first() != endItems.first() )
353 {
354
355 bool Inheritance = KernelWrapper::GetInstance().GetInheritanceMode();
356 SchemaGraphicObject * startItem = qgraphicsitem_cast<SchemaGraphicObject *> (
357 startItems.first() );
358 SchemaGraphicObject * endItem = qgraphicsitem_cast<SchemaGraphicObject *> (
359 endItems.first() );
360
361 if ( Inheritance )
362 {
363 startItem->GetClass()->add_super_class ( endItem->GetClassName().toStdString() );
365 SchemaGraphicArrow * newArrow = new SchemaGraphicArrow ( startItem, endItem, Inheritance,
366 true, "", "" );
367 startItem->AddArrow ( newArrow );
368 endItem->AddArrow ( newArrow );
369 newArrow->setZValue ( -1000.0 );
370 addItem ( newArrow );
371 //newArrow->SetLabelScene(this);
372 newArrow->UpdatePosition();
373 }
374 else
375 {
376 SchemaRelationshipEditor * Editor = new SchemaRelationshipEditor (
377 startItem->GetClass(), endItem->GetClassName() );
378 connect ( Editor, SIGNAL ( MakeGraphConnection ( QString, QString, QString ) ), this,
379 SLOT ( DrawArrow ( QString, QString, QString ) ) );
380 Editor->show();
381 }
382 }
383 }
384
385 line = nullptr;
386 QGraphicsScene::mouseReleaseEvent ( mouseEvent );
387}
