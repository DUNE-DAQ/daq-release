67std::vector<const coredal::DaqModule*>
68DFApplication::generate_modules(conffwk::Configuration* confdb,
69 const std::string& dbfile,
70 const coredal::Session* session) const
71{
72 std::vector<const coredal::DaqModule*> modules;
73
74 // Containers for module specific config objects for output/input
75 // Prepare TRB output objects
76 std::vector<const conffwk::ConfigObject*> trbOutputObjs;
77
78 // -- First, we process expected Queue and Network connections and create their objects.
79
80 // Process the queue rules looking for the TriggerRecord queue between TRB and DataWriter
81 const QueueDescriptor* trQDesc = nullptr;
82 for (auto rule : get_queue_rules()) {
83 auto destination_class = rule->get_destination_class();
84 if (destination_class == "DataWriter") {
85 trQDesc = rule->get_descriptor();
86 }
87 }
88 if (trQDesc == nullptr) { // BadConf if no descriptor between TRB and DataWriter
89 throw(BadConf(ERS_HERE, "Could not find queue descriptor rule for TriggerRecords!"));
90 }
91 // Create queue connection config object
92 conffwk::ConfigObject trQueueObj;
93 std::string trQueueUid(trQDesc->get_uid_base() + UID());
94 confdb->create(dbfile, "Queue", trQueueUid, trQueueObj);
95 fill_queue_object_from_desc(trQDesc, trQueueObj);
96 // Place trigger record queue object into vector of output objs of TRB module
97 trbOutputObjs.push_back(&trQueueObj);
98
99 // Process the network rules looking for the Fragments and TriggerDecision inputs for TRB
100 const NetworkConnectionDescriptor* fragNetDesc = nullptr;
101 const NetworkConnectionDescriptor* trigdecNetDesc = nullptr;
102 const NetworkConnectionDescriptor* tokenNetDesc = nullptr;
103 for (auto rule : get_network_rules()) {
104 auto endpoint_class = rule->get_endpoint_class();
105 auto descriptor = rule->get_descriptor();
106 auto data_type = descriptor->get_data_type();
107 if (data_type == "Fragment") {
108 fragNetDesc = rule->get_descriptor();
109 } else if (data_type == "TriggerDecision") {
110 trigdecNetDesc = rule->get_descriptor();
111 } else if (data_type == "TriggerDecisionToken") {
112 tokenNetDesc = rule->get_descriptor();
113 }
114 }
115 if (fragNetDesc == nullptr) { // BadConf if no descriptor for Fragments into TRB
116 throw(BadConf(ERS_HERE, "Could not find network descriptor rule for input Fragments!"));
117 }
118 if (trigdecNetDesc == nullptr) { // BadCond if no descriptor for TriggerDecisions into TRB
119 throw(BadConf(ERS_HERE, "Could not find network descriptor rule for input TriggerDecisions!"));
120 }
121 if (tokenNetDesc == nullptr) { // BadCond if no descriptor for Tokens out of DataWriter
122 throw(BadConf(ERS_HERE, "Could not find network descriptor rule for output TriggerDecisionTokens!"));
123 }
124 if (get_source_id() == nullptr) {
125 throw(BadConf(ERS_HERE, "Could not retrieve SourceIDConf"));
126 }
127 // Create network connection config object
128 conffwk::ConfigObject fragNetObj;
129 conffwk::ConfigObject trigdecNetObj;
130 conffwk::ConfigObject tokenNetObj;
131 std::string fragNetUid = fragNetDesc->get_uid_base() + UID();
132 std::string trigdecNetUid = trigdecNetDesc->get_uid_base() + UID();
133 std::string tokenNetUid = tokenNetDesc->get_uid_base();
134 confdb->create(dbfile, "NetworkConnection", fragNetUid, fragNetObj);
135 confdb->create(dbfile, "NetworkConnection", trigdecNetUid, trigdecNetObj);
136 confdb->create(dbfile, "NetworkConnection", tokenNetUid, tokenNetObj);
137 fill_netconn_object_from_desc(fragNetDesc, fragNetObj);
138 fill_netconn_object_from_desc(trigdecNetDesc, trigdecNetObj);
139 fill_netconn_object_from_desc(tokenNetDesc, tokenNetObj);
140
141 // Process special Network rules!
142 // Looking for DataRequest rules from ReadoutAppplications in current Session
143 auto sessionApps = session->get_all_applications();
144 std::vector<conffwk::ConfigObject> dreqNetObjs;
145 for (auto app : sessionApps) {
146 auto roapp = app->cast<appdal::ReadoutApplication>();
147 if (roapp != nullptr) {
148 TLOG() << "Readout app in session: " << roapp;
149 auto roQRules = roapp->get_network_rules();
150 for (auto rule : roQRules) {
151 auto descriptor = rule->get_descriptor();
152 auto data_type = descriptor->get_data_type();
153 if (data_type == "DataRequest") {
154 dreqNetObjs.emplace_back();
155 std::string dreqNetUid(descriptor->get_uid_base() + roapp->UID());
156 confdb->create(dbfile, "NetworkConnection", dreqNetUid, dreqNetObjs.back());
157 fill_netconn_object_from_desc(descriptor, dreqNetObjs.back());
158 } // If network rule has DataRequest type of data
159 } // Loop over ReadoutApps network rules
160 } // if app is ReadoutApplication
161 } // loop over Session specific Apps
162
163 // Get pointers to objects here, after vector has been filled so they don't move on us
164 for (auto& obj : dreqNetObjs) {
165 trbOutputObjs.push_back(&obj);
166 }
167
168 // -- Second, we create the Module objects and assign their configs, with the precreated
169 // -- connection config objects above.
170
171 // Get TRB Config Object
172 auto trbConf = get_trb();
173 if (trbConf == 0) {
174 throw(BadConf(ERS_HERE, "No DataWriter or TRB configuration given"));
175 }
176 auto trbConfObj = trbConf->config_object();
177 trbConfObj.set_by_val<uint32_t>("source_id", get_source_id()->get_sid());
178 // Prepare TRB Module Object and assign its Config Object.
179 conffwk::ConfigObject trbObj;
180 std::string trbUid(UID() + "-trb");
181 confdb->create(dbfile, "TriggerRecordBuilder", trbUid, trbObj);
182 trbObj.set_obj("configuration", &trbConfObj);
183 trbObj.set_objs("inputs", { &trigdecNetObj, &fragNetObj });
184 trbObj.set_objs("outputs", trbOutputObjs);
185 // Push TRB Module Object from confdb
186 modules.push_back(confdb->get<TriggerRecordBuilder>(trbUid));
187
188 // Get DataWriter Config Object (only one for now, maybe more later?)
189 auto dwrConf = get_data_writer();
190 if (dwrConf == 0) {
191 throw(BadConf(ERS_HERE, "No DataWriter or TRB configuration given"));
192 }
193 auto fnParamsObj = dwrConf->get_data_store_params()->get_filename_params()->config_object();
194 fnParamsObj.set_by_val<std::string>("writer_identifier", UID() + "_datawriter-1");
195 auto dwrConfObj = dwrConf->config_object();
196 // Prepare DataWriter Module Object and assign its Config Object.
197 conffwk::ConfigObject dwrObj;
198 std::string dwrUid(UID() + "-dw-1");
199 confdb->create(dbfile, "DataWriter", dwrUid, dwrObj);
200 dwrObj.set_obj("configuration", &dwrConfObj);
201 dwrObj.set_objs("inputs", { &trQueueObj });
202 dwrObj.set_objs("outputs", { &tokenNetObj });
203 // Push DataWriter Module Object from confdb
204 modules.push_back(confdb->get<DataWriter>(dwrUid));
205
206 return modules;
207}
