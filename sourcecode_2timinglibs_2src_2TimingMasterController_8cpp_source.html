215void
216TimingMasterController::endpoint_scan(std::atomic<bool>& running_flag)
217{
218
219 std::ostringstream starting_stream;
220 starting_stream << ": Starting endpoint_scan() method.";
221 TLOG_DEBUG(0) << get_name() << starting_stream.str();
222
223 while (running_flag.load() && m_endpoint_scan_period) {
224
225 timingcmd::TimingHwCmd hw_cmd =
226 construct_master_hw_cmd( "master_endpoint_scan");
227
228 timingcmd::TimingMasterEndpointScanPayload cmd_payload;
229 cmd_payload.endpoints = m_monitored_endpoint_locations;
230
231 hw_cmd.payload = cmd_payload;
232 send_hw_cmd(std::move(hw_cmd));
233
234 ++(m_sent_hw_command_counters.at(3).atomic);
235 if (m_endpoint_scan_period)
236 {
237 auto prev_gather_time = std::chrono::steady_clock::now();
238 auto next_gather_time = prev_gather_time + std::chrono::milliseconds(m_endpoint_scan_period);
239
240 // check running_flag periodically
241 auto slice_period = std::chrono::microseconds(10000);
242 auto next_slice_gather_time = prev_gather_time + slice_period;
243
244 bool break_flag = false;
245 while (next_gather_time > next_slice_gather_time + slice_period) {
246 if (!running_flag.load()) {
247 TLOG_DEBUG(0) << "while waiting to send delays, negative run gatherer flag detected.";
248 break_flag = true;
249 break;
250 }
251 std::this_thread::sleep_until(next_slice_gather_time);
252 next_slice_gather_time = next_slice_gather_time + slice_period;
253 }
254 if (break_flag == false) {
255 std::this_thread::sleep_until(next_gather_time);
256 }
257 }
258 else
259 {
260 TLOG() << "m_endpoint_scan_period is 0 and send delays thread is running! breaking loop!";
261 break;
262 }
263 }
264
265 std::ostringstream exiting_stream;
266 exiting_stream << ": Exiting endpoint_scan() method. Received " << m_sent_hw_command_counters.at(3).atomic.load()
267 << " commands";
268 TLOG_DEBUG(0) << get_name() << exiting_stream.str();
269}
