66void
67FIBIONode::reset(int32_t fanout_mode, const std::string& clock_config_file) const {
68
69 // Soft reset
70 write_soft_reset_register();
71
72 millisleep(1000);
73
74 // Reset I2C
75 getNode("csr.ctrl.rstb_i2c").write(0x1);
76 getNode("csr.ctrl.rstb_i2c").write(0x0);
77
78 getNode("csr.ctrl.rst").write(0x1);
79 getNode("csr.ctrl.rst").write(0x0);
80
81 getClient().dispatch();
82
83 const CarrierType carrier_type = convert_value_to_carrier_type(read_carrier_type());
84
85 if (carrier_type == kCarrierEnclustraA35) {
86 // enclustra i2c switch stuff
87 try {
88 getNode<I2CMasterNode>(m_uid_i2c_bus).get_slave("AX3_Switch").write_i2c(0x01, 0x7f);
89 } catch (const std::exception& e) {
90 ers::warning(EnclustraSwitchFailure(ERS_HERE, e));
91 }
92 }
93
94 // Configure I2C IO expanders
95 auto ic_10 = get_i2c_device<I2CExpanderSlave>(m_uid_i2c_bus, "Expander1");
96 auto ic_23 = get_i2c_device<I2CExpanderSlave>(m_uid_i2c_bus, "Expander2");
97
98 // Bank 0
99 ic_10->set_inversion(0, 0x00);
100
101 // all out, sfp tx disable
102 ic_10->set_io(0, 0x00);
103 // sfp laser on by default
104 ic_10->set_outputs(0, 0x00);
105
106 // Bank 1
107 ic_10->set_inversion(1, 0x00);
108 // all inputs, sfp fault
109 ic_10->set_io(1, 0xff);
110
111 // Bank 0
112 ic_23->set_inversion(0, 0x00);
113 // pin 0 - out: pll rst, pins 1-4 pll and cdr flags
114 ic_23->set_io(0, 0xfe);
115 ic_23->set_outputs(0, 0x01);
116
117 // Bank 1
118 ic_23->set_inversion(1, 0x00);
119 // all inputs, sfp los
120 ic_23->set_io(1, 0xff);
121
122 // reset pll via I2C IO expanders
123 reset_pll();
124
125 // Find the right pll config file
126 std::string clock_config_path = get_full_clock_config_file_path(clock_config_file, fanout_mode);
127 TLOG() << "PLL configuration file : " << clock_config_path;
128
129 // Upload config file to PLL
130 configure_pll(clock_config_path);
131
132 //getNode("csr.ctrl.inmux").write(0);
133 //getClient().dispatch();
134
135// To be removed from firmware address maps also
136// getNode("csr.ctrl.rst_lock_mon").write(0x1);
137// getNode("csr.ctrl.rst_lock_mon").write(0x0);
138// getClient().dispatch();
139
140 TLOG() << "Reset done";
141}

254void
255FIBIONode::switch_sfp_tx(uint32_t sfp_id, bool turn_on) const { // NOLINT(build/unsigned)
256 validate_sfp_id(sfp_id);
257
258 auto ic_10 = get_i2c_device<I2CExpanderSlave>(m_uid_i2c_bus, "Expander1");
259 uint8_t current_sfp_tx_control_flags = ic_10->read_outputs_config(0); // NOLINT(build/unsigned)
260
261 uint8_t new_sfp_tx_control_flags; // NOLINT(build/unsigned)
262 if (turn_on)
263 {
264 new_sfp_tx_control_flags = current_sfp_tx_control_flags & ~(1UL << sfp_id);
265 }
266 else
267 {
268 new_sfp_tx_control_flags = current_sfp_tx_control_flags | (1UL << sfp_id);
269 }
270
271 ic_10->set_outputs(0, new_sfp_tx_control_flags);
272}

334void
335FIBIONode::get_info(opmonlib::InfoCollector& ci, int level) const
336{
337
338 if (level >= 2) {
339 timinghardwareinfo::TimingPLLMonitorData pll_mon_data;
340 get_pll()->get_info(pll_mon_data);
341 ci.add(pll_mon_data);
342
343 for (uint i=0; i < 8; ++i)
344 {
345 opmonlib::InfoCollector sfp_ic;
346
347 std::string sfp_i2c_bus = "i2c_sfp" + std::to_string(i);
348 auto sfp = get_i2c_device<I2CSFPSlave>(sfp_i2c_bus, "SFP_EEProm");
349
350 try
351 {
352 sfp->get_info(sfp_ic, level);
353 }
354 catch (timing::SFPUnreachable& e)
355 {
356 // It is valid that an SFP may not be installed, currently no good way of knowing whether they it should be
357 TLOG_DEBUG(2) << "Failed to communicate with SFP " << i << " on I2C switch channel " << (1UL << i) << " on i2c bus" << m_sfp_i2c_buses.at(0);
358 continue;
359 }
360 ci.add("sfp_"+std::to_string(i),sfp_ic);
361 }
362 }
363 if (level >= 1) {
364 timinghardwareinfo::TimingFIBMonitorData mon_data;
365 this->get_info(mon_data);
366 ci.add(mon_data);
367 }
368}
