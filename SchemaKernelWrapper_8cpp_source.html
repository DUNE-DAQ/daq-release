114bool dbse::KernelWrapper::AnyClassReferenceThis ( OksClass * SchemaClass )
115{
116 std::vector<OksClass *> ClassList;
117 GetClassList ( ClassList );
118
119 for ( OksClass * ClassInfo : ClassList )
120 {
121 const std::list<OksRelationship *> * RelationshipList = ClassInfo->direct_relationships();
122
123 if ( RelationshipList != nullptr )
124 {
125 for ( OksRelationship * RelationshipInfo : *RelationshipList )
126 {
127 if ( RelationshipInfo->get_class_type()->get_name() == SchemaClass->get_name() )
128 {
129 return true;
130 }
131 }
132 }
133 }
134
135 return false;
136}

138QString dbse::KernelWrapper::GetCardinalityStringRelationship (
139 OksRelationship * SchemaRelationship ) const
140{
141 QString RelationshipLowCc (
142 OksRelationship::card2str ( SchemaRelationship->get_low_cardinality_constraint() ) );
143 QString RelationshipHighCc (
144 OksRelationship::card2str ( SchemaRelationship->get_high_cardinality_constraint() ) );
145
146 if ( RelationshipLowCc == "zero" )
147 {
148 RelationshipLowCc = "0";
149 }
150 else if ( RelationshipLowCc == "one" )
151 {
152 RelationshipLowCc = "1";
153 }
154 else if ( RelationshipLowCc == "many" )
155 {
156 RelationshipLowCc = "n";
157 }
158
159 if ( RelationshipHighCc == "zero" )
160 {
161 RelationshipHighCc = "0";
162 }
163 else if ( RelationshipHighCc == "one" )
164 {
165 RelationshipHighCc = "1";
166 }
167 else if ( RelationshipHighCc == "many" )
168 {
169 RelationshipHighCc = "n";
170 }
171
172 return QString ( RelationshipLowCc + ':' + RelationshipHighCc );
173}
