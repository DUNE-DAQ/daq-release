84Application::execute(const dataobj_t& cmd_data)
85{
86
87 auto rc_cmd = cmd_data.get<rcif::cmd::RCCommand>();
88 std::string cmdname = rc_cmd.id;
89 if (!is_cmd_valid(cmd_data)) {
90 throw InvalidCommand(ERS_HERE, cmdname, get_state(), m_error.load(), m_busy.load());
91 }
92
93 m_busy.store(true);
94
95 if (cmdname == "start") {
96 auto cmd_obj = rc_cmd.data.get<cmd::CmdObj>();
97
98 for (const auto& addressed : cmd_obj.modules) {
99 dataobj_t startpars = addressed.data;
100 auto rc_startpars = startpars.get<rcif::cmd::StartParams>();
101 m_runinfo.runno = rc_startpars.run;
102 break;
103 }
104
105 m_run_start_time = std::chrono::steady_clock::now();
106 ;
107 m_runinfo.running = true;
108 m_runinfo.runtime = 0;
109 } else if (cmdname == "stop") {
110 m_runinfo.running = false;
111 m_runinfo.runno = 0;
112 m_runinfo.runtime = 0;
113 }
114
115 try {
116 m_mod_mgr.execute(get_state(), cmdname, rc_cmd.data);
117 m_busy.store(false);
118 if (rc_cmd.exit_state != "ANY")
119 set_state(rc_cmd.exit_state);
120 } catch (ers::Issue& ex) {
121 m_busy.store(false);
122 m_error.store(true);
123 throw;
124 }

164bool
165Application::is_cmd_valid(const dataobj_t& cmd_data)
166{
167 if (m_busy.load() || m_error.load())
168 return false;
169
170 std::string state = get_state();
171 std::string entry_state = cmd_data.get<rcif::cmd::RCCommand>().entry_state;
172 if (entry_state == "ANY" || state == entry_state)
173 return true;
174
175 return false;
