177void
178IfaceWrapper::setup_flow_steering()
179{
180 // Flow steering setup
181 TLOG() << "Configuring Flow steering rules for iface=" << m_iface_id;
182 struct rte_flow_error error;
183 struct rte_flow *flow;
184 TLOG() << "Attempt to flush previous flow rules...";
185 rte_flow_flush(m_iface_id, &error);
186#warning RS: FIXME -> Check for flow flush return!
187 for (auto const& [lcoreid, rxqs] : m_rx_core_map) {
188 for (auto const& [rxqid, srcip] : rxqs) {
189 // Put the IP numbers temporarily in a vector, so they can be converted easily to uint32_t
190 TLOG() << "Creating flow rule for src_ip=" << srcip << " assigned to rxq=" << rxqid;
191 size_t ind = 0, current_ind = 0;
192 std::vector<uint8_t> v;
193 for (int i = 0; i < 4; ++i) {
194 v.push_back(std::stoi(srcip.substr(current_ind, srcip.size() - current_ind), &ind));
195 current_ind += ind + 1;
196 }
197
198 flow = generate_ipv4_flow(m_iface_id, rxqid,
199 RTE_IPV4(v[0], v[1], v[2], v[3]), 0xffffffff, 0, 0, &error);
200
201 if (not flow) { // ers::fatal
202 TLOG() << "Flow can't be created for " << rxqid
203 << " Error type: " << (unsigned)error.type
204 << " Message: " << error.message;
205 ers::fatal(dunedaq::readoutlibs::InitializationError(
206 ERS_HERE, "Couldn't create Flow API rules!"));
207 rte_exit(EXIT_FAILURE, "error in creating flow");
208 }
209 }
210 }
211
212 return;

268IfaceWrapper::get_info(opmonlib::InfoCollector& ci, int level)
269{
270
271 nicreaderinfo::EthStats s;
272 s.ipackets = m_iface_xstats.m_eth_stats.ipackets;
273 s.opackets = m_iface_xstats.m_eth_stats.opackets;
274 s.ibytes = m_iface_xstats.m_eth_stats.ibytes;
275 s.obytes = m_iface_xstats.m_eth_stats.obytes;
276 s.imissed = m_iface_xstats.m_eth_stats.imissed;
277 s.ierrors = m_iface_xstats.m_eth_stats.ierrors;
278 s.oerrors = m_iface_xstats.m_eth_stats.oerrors;
279 s.rx_nombuf = m_iface_xstats.m_eth_stats.rx_nombuf;
280 ci.add(s);
281
282 // Empty stat JSON placeholder
283 nlohmann::json stat_json;
284
285 // Poll stats from HW
286 m_iface_xstats.poll();
287
288 // Build JSON from values
289 for (int i = 0; i < m_iface_xstats.m_len; ++i) {
290 stat_json[m_iface_xstats.m_xstats_names[i].name] = m_iface_xstats.m_xstats_values[i];
291 }
292
293 // Reset HW counters
294 m_iface_xstats.reset_counters();
295
296 // Convert JSON to NICReaderInfo struct
297 nicreaderinfo::EthXStats xs;
298 nicreaderinfo::from_json(stat_json, xs);
299
300 // Push to InfoCollector
301 ci.add(xs);
302 TLOG_DEBUG(TLVL_WORK_STEPS) << "opmonlib::InfoCollector object passed by reference to IfaceWrapper::get_info"
303 << " -> Result looks like the following:\n" << ci.get_collected_infos();
304
305 for( const auto& [src_rx_q,_] : m_num_frames_rxq) {
306 nicreaderinfo::QueueStats qs;
307 qs.packets_received = m_num_frames_rxq[src_rx_q].load();
308 qs.bytes_received = m_num_bytes_rxq[src_rx_q].load();
309 qs.full_rx_burst = m_num_full_bursts[src_rx_q].load();
310 qs.max_burst_size = m_max_burst_size[src_rx_q].exchange(0);
311
312 opmonlib::InfoCollector queue_ci;
313 queue_ci.add(qs);
314
315 ci.add(fmt::format("queue_{}", src_rx_q), queue_ci);
316 }
317}

331void
332IfaceWrapper::handle_eth_payload(int src_rx_q, char* payload, std::size_t size)
333{
334 // Get DAQ Header and its StreamID
335 auto* daq_header = reinterpret_cast<dunedaq::detdataformats::DAQEthHeader*>(payload);
336 auto src_id = m_stream_to_source_id[src_rx_q][(unsigned)daq_header->stream_id];
337
338 if ( auto src_it = m_sources.find(src_id); src_it != m_sources.end()) {
339 src_it->second->handle_payload(payload, size);
340 } else {
341 // Really bad -> unexpeced StreamID in UDP Payload.
342 // This check is needed in order to avoid dynamically add thousands
343 // of Sources on the fly, in case the data corruption is extremely severe.
344 if (m_num_unexid_frames.count(src_id) == 0) {
345 m_num_unexid_frames[src_id] = 0;
346 }
347 m_num_unexid_frames[src_id]++;
348 }
349}
