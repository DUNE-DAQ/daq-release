181void
182PartitionNode::start(uint32_t timeout /*milliseconds*/) const // NOLINT(build/unsigned)
183{
184
185 // Disable triggers (just in case)
186 getNode("csr.ctrl.trig_en").write(0);
187 // Disable the buffer
188 getNode("csr.ctrl.buf_en").write(0);
189 getClient().dispatch();
190 // Re-enable the buffer (flushes it)
191 getNode("csr.ctrl.buf_en").write(1);
192 getClient().dispatch();
193
194 // Set the run bit and wait for it to be acknowledged
195 getNode("csr.ctrl.run_req").write(1);
196 getClient().dispatch();
197
198 std::chrono::steady_clock::time_point start = std::chrono::steady_clock::now();
199
200 while (true) {
201 auto in_run = getNode("csr.stat.in_run").read();
202 getClient().dispatch();
203
204 if (in_run)
205 break;
206
207 std::chrono::steady_clock::time_point end = std::chrono::steady_clock::now();
208 if ((end - start) > std::chrono::milliseconds(timeout)) {
209 throw RunRequestTimeoutExpired(ERS_HERE, timeout);
210 }
211 std::this_thread::sleep_for(std::chrono::milliseconds(100));
212 }
213}
