103void
104TimingController::send_hw_cmd(timingcmd::TimingHwCmd&& hw_cmd)
105{
106 if (!m_hw_command_sender)
107 {
108 throw QueueIsNullFatalError(ERS_HERE, get_name(), m_hw_command_out_connection);
109 }
110 try {
111 m_hw_command_sender->send(std::move(hw_cmd), m_hw_cmd_out_timeout);
112 } catch (const dunedaq::iomanager::TimeoutExpired& excpt) {
113 std::ostringstream oss_warn;
114 oss_warn << "push to output queue \"" << m_hw_command_out_connection << "\"";
115 ers::warning(dunedaq::iomanager::TimeoutExpired(
116 ERS_HERE,
117 get_name(),
118 oss_warn.str(),
119 std::chrono::duration_cast<std::chrono::milliseconds>(m_hw_cmd_out_timeout).count()));
120 }
