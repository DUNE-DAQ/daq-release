20 frame = inspect.currentframe()
21 while hasattr(frame, "f_code"):
22 co = frame.f_code
23 filename = os.path.normcase(co.co_filename)
24 if 'ERSPublisher.py' not in filename:
25 # Found the frame of the original caller
26 break

57def create_issue(message, name="GenericPythonIssue", severity=SeverityLevel.INFO.name, cause=None):
58 """Create an ERS IssueChain with minimal user input."""
59 # This creates an issue chain with a given name, message, and severity
60 # The message can be a Python exception; in that case the name is also overwritten, with the name of the exception
61 # The cause can be another issue (chain or simple) or an exception
62
63 current_time = time.time_ns()
64
65 if isinstance(message,Exception):
66 # If the issue is created from an exception, set the name and inheritance
67 issue = exception_to_issue(message,severity) # Use the existing function to create an issue from the exception
68 issue.time = current_time # The time is overwritten as this is the time of the call of the function
69 else:
70 # For non-exception issues, continue as normal
71 inheritance_list = ["PythonIssue", name]
72 issue = ersissue.SimpleIssue(
73 context=generate_context(),
74 name=name,
75 message=message,
76 time=current_time,
77 severity=str(severity),
78 inheritance=inheritance_list
79 )
80
81 issue_chain = ersissue.IssueChain(
82 final=issue,
83 session=os.getenv('DUNEDAQ_PARTITION', 'Unknown'),
84 application="python"
85 )
86
87 # Process the cause and add to issue_chain.causes, but not to issue.inheritance
88 if cause:
89 if isinstance(cause, Exception):
90 # Convert exception to a SimpleIssue and append to causes of issue_chain
91 cause_issue = exception_to_issue(cause,severity)
92 issue_chain.causes.extend([cause_issue])
93 elif isinstance(cause, ersissue.SimpleIssue):
94 # Add the cause directly to the causes of issue_chain
95 issue_chain.causes.extend([cause])
96 elif isinstance(cause, ersissue.IssueChain):
97 # Set the final cause of the existing chain as the first cause of the new chain
98 issue_chain.causes.append(cause.final)
99 issue_chain.causes.extend(cause.causes)
100
101 return issue_chain
