201void
202MasterNode::apply_endpoint_delay(uint32_t address, // NOLINT(build/unsigned)
203 uint32_t coarse_delay, // NOLINT(build/unsigned)
204 uint32_t fine_delay, // NOLINT(build/unsigned)
205 uint32_t /*phase_delay*/, // NOLINT(build/unsigned)
206 bool measure_rtt,
207 bool control_sfp) const
208{
209
210 auto global = getNode<MasterGlobalNode>("global");
211 auto echo = getNode<EchoMonitorNode>("echo_mon");
212
213 if (measure_rtt) {
214 if (control_sfp) {
215 // Switch off all TX SFPs
216 // switch_endpoint_sfp(0xffff, false);
217
218 // Turn on the current target
219 switch_endpoint_sfp(address, true);
220
221 millisleep(100);
222 }
223
224 try
225 {
226 global.enable_upstream_endpoint();
227 }
228 catch (const timing::ReceiverNotReady& e)
229 {
230 if (control_sfp) {
231 switch_endpoint_sfp(address, false);
232 }
233 throw e;
234 }
235
236 uint64_t endpoint_rtt = echo.send_echo_and_measure_delay(); // NOLINT(build/unsigned)
237 TLOG() << "Pre delay adjustment RTT: " << format_reg_value(endpoint_rtt, 10);
238 }
239
240 uint32_t sequence = 0xab;
241 uint32_t address_mode = 1;
242
243 std::vector<uint32_t> tx_packet = { address & 0xff,
244 address >> 8UL,
245 sequence,
246
247 // packet to write coarse delay
248 (0x1 << 7UL) | 0x72, // write transaction on 0x72
249 (address_mode << 7UL) | 0x1, // transaction length of 0x1
250 ((fine_delay & 0xf) << 4UL) | coarse_delay & 0xf,
251
252 // packet to write fine delay
253 (0x1 << 7UL) | 0x73, // write transaction on 0x73
254 (address_mode << 7UL) | 0x1, // transaction length of 0x1
255 (fine_delay >> 4UL) & 0xff,
256
257 // packet to set skew done
258 (0x1 << 7UL) | 0x70, // write transaction on 0x70
259 (address_mode << 7UL) | 0x1, // transaction length of 0x1
260 0x3, // deskew done
261
262 // packet to resync
263 (0x1 << 7UL) | 0x70, // write transaction on 0x70
264 (address_mode << 7UL) | 0x1, // transaction length of 0x1
265 0x4, // resync
266 };
267
268 tx_packet.back() = tx_packet.back() | (0x1 << 8UL);
269
270 transmit_async_packet(tx_packet, -1);
271
272 if (measure_rtt) {
273 try
274 {
275 global.enable_upstream_endpoint();
276 }
277 catch (const timing::ReceiverNotReady& e)
278 {
279 if (control_sfp)
280 {
281 switch_endpoint_sfp(address, false);
282 }
283 throw e;
284 }
285
286 uint64_t endpoint_rtt = echo.send_echo_and_measure_delay(); // NOLINT(build/unsigned)
287 TLOG() << "Post delay adjustment RTT: " << format_reg_value(endpoint_rtt, 10);
288
289 if (control_sfp)
290 switch_endpoint_sfp(address, false);
291 }

390std::vector<uint32_t>
391MasterNode::transmit_async_packet(const std::vector<uint32_t>& packet, int timeout) const
392{
393 // TODO: check for valid packet
394
395 reset_sub_nodes(getNode("acmd_buf.txbuf"));
396
397 TLOG_DEBUG(11) << "tx packet: ";
398 for (auto t : packet)
399 TLOG_DEBUG(11) << std::hex << "0x" << t;
400
401 getNode("acmd_buf.txbuf").writeBlock(packet);
402 getClient().dispatch();
403
404 // we do not expect a reply
405 if (timeout < 0)
406 {
407 std::vector<uint32_t> empty_vector;
408 return empty_vector;
409 }
410
411 uhal::ValWord<uint32_t> buffer_ready; // NOLINT(build/unsigned)
412 uhal::ValWord<uint32_t> buffer_timeout; // NOLINT(build/unsigned)
413
414 // start time counting
415 auto start = std::chrono::high_resolution_clock::now();
416
417 // Wait for the buffer to be happy
418 while (true) {
419
420 buffer_ready = getNode("acmd_buf.stat.ready").read();
421 buffer_timeout = getNode("acmd_buf.stat.timeout").read();
422 getClient().dispatch();
423
424 TLOG_DEBUG(10) << "async buffer ready: 0x" << buffer_ready.value() << ", timeout: " << buffer_timeout.value();
425
426 if (buffer_timeout)
427 throw VLCommandReplyTimeout(ERS_HERE);
428
429 if (buffer_ready)
430 break;
431
432 auto now = std::chrono::high_resolution_clock::now();
433 auto us_since_start = std::chrono::duration_cast<std::chrono::microseconds>(now - start);
434
435 if (us_since_start.count() > timeout)
436 throw VLCommandReplyBufferFlagTimeout(ERS_HERE, timeout);
437
438 std::this_thread::sleep_for(std::chrono::microseconds(50));
439 }
440
441 auto rx_packet = getNode("acmd_buf.rxbuf").readBlock(0x20);
442 getClient().dispatch();
443
444 if (rx_packet.at(0) != 0xff || rx_packet.at(1) != 0xff || rx_packet.at(2) != packet.at(2))
445 {
446 ers::warning(InvalidVLCommandReplyPacket(ERS_HERE, rx_packet.at(0), rx_packet.at(1), rx_packet.at(2)));
447 }
448
449 TLOG_DEBUG(11) << "async result: ";
450 for (auto r : rx_packet)
451 TLOG_DEBUG(11) << std::hex << "0x" << r;
452
453 return rx_packet.value();
454}

458void
459MasterNode::write_endpoint_data(uint16_t endpoint_address, uint8_t reg_address, std::vector<uint8_t> data, bool address_mode) const
460{
461 auto data_length = data.size();
462 if (data_length > 0x3f || data_length == 0)
463 {
464 TLOG() << "invalid data length";
465 }
466
467 // TODO make sequence a function argument?
468 uint32_t sequence = 0xab;
469
470 std::vector<uint32_t> tx_packet = { static_cast<uint32_t>(endpoint_address & 0xff),
471 static_cast<uint32_t>(endpoint_address >> 8UL),
472 sequence,
473 // bit 7 = 1 -> write
474 static_cast<uint32_t>((0x1 << 7UL) | reg_address),
475 static_cast<uint32_t>((address_mode << 7UL) | (0x3f & data_length))
476 };
477 tx_packet.insert(tx_packet.end(), data.begin(), data.end());
478 tx_packet.back() = tx_packet.back() | (0x1 << 8UL);
479
480 auto result = transmit_async_packet(tx_packet);
481}

485std::vector<uint32_t>
486MasterNode::read_endpoint_data(uint16_t endpoint_address, uint8_t reg_address, uint8_t data_length, bool address_mode) const
487{
488 if (data_length > 0x3f || data_length == 0)
489 {
490 TLOG() << "invalid data length";
491 // TODO throw something
492 }
493
494 // TODO make sequence a function argument?
495 uint32_t sequence = 0xab;
496 std::vector<uint32_t> tx_packet = { static_cast<uint32_t>(endpoint_address & 0xff),
497 static_cast<uint32_t>(endpoint_address >> 8UL),
498 sequence,
499 // bit 7 = 0 -> read
500 reg_address,
501 static_cast<uint32_t>((0x1 << 8UL) | (address_mode << 7UL) | (0x3f & data_length))
502 };
503
504 auto result = transmit_async_packet(tx_packet);
505
506 // get parts we actually want
507 std::vector<uint32_t> result_data (result.begin()+3, result.begin()+3+data_length);
508
509 // strip off the bit 8 which is high for last byte
510 result_data.back() = result_data.back() & 0xff;
511
512 return result_data;
513}

533timingfirmware::EndpointCheckResult
534MasterNode::scan_endpoint(uint16_t endpoint_address, bool control_sfp) const
535{
536 timingfirmware::EndpointCheckResult result;
537 auto global = getNode<MasterGlobalNode>("global");
538 auto echo = getNode<EchoMonitorNode>("echo_mon");
539
540 timingfirmware::EndpointCheckResult endpoint_result;
541 endpoint_result.address = endpoint_address;
542
543 // is endpoint sfp switched on?
544 // are any relevant muxes set to correct channel?
545 if (control_sfp)
546 {
547 switch_endpoint_sfp(endpoint_address, true);
548
549 millisleep(100);
550 }
551
552 try
553 {
554 global.enable_upstream_endpoint();
555 }
556 catch (const timing::ReceiverNotReady& e)
557 {
558 switch_endpoint_sfp(endpoint_address, false);
559
560 ers::error(MonitoredEndpointDead(ERS_HERE, endpoint_address));
561
562 return endpoint_result;
563 }
564
565 endpoint_result.alive = true;
566 endpoint_result.round_trip_time = echo.send_echo_and_measure_delay();
567 TLOG_DEBUG(5) << "Endpoint at address " << endpoint_address << " alive. RTT: " << endpoint_result.round_trip_time;
568
569 auto ept_state = read_endpoint_data(endpoint_address, 0x71, 0x1, 0x1).at(0) & 0xf;
570 TLOG_DEBUG(5) << "Endpoint at address " << endpoint_address << " state: 0x" << std::hex << ept_state;
571 endpoint_result.state = ept_state;
572
573 if (ept_state == 0x6)
574 {
575 TLOG_DEBUG(5) << "Endpoint at address " << endpoint_address << ", applying delays of: " << 0x0;
576
577 apply_endpoint_delay(endpoint_address, 0x0, 0x0, 0x0, false, false);
578
579 endpoint_result.applied_delay = 0x0;
580
581 auto ept_state_after_delays = read_endpoint_data(endpoint_address, 0x71, 0x1, 0x1).at(0) & 0xf;
582 TLOG_DEBUG(5) << "Endpoint at address " << endpoint_address << ", state after delays apply: " << ept_state_after_delays;
583 endpoint_result.state_after_delay_apply = ept_state_after_delays;
584
585 endpoint_result.round_trip_time_after_delay_apply = echo.send_echo_and_measure_delay();
586 TLOG_DEBUG(5) << "Endpoint at address " << endpoint_address << ", RTT after delays apply: " << endpoint_result.round_trip_time_after_delay_apply;
587 }
588 else if (ept_state == 0x7 || ept_state == 0x8)
589 {
590 TLOG_DEBUG(5) << "Endpoint at address " << endpoint_address << ", delays not needed";
591 }
592 else
593 {
594 ers::error(MonitoredEndpointUnexpectedState(ERS_HERE, endpoint_address, ept_state));
595 }
596
597 if (control_sfp)
598 {
599 switch_endpoint_sfp(endpoint_address, false);
600 }
601
602 return endpoint_result;
603}
