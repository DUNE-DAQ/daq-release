68SpillInterfaceNode::enable_fake_spills(uint32_t cycle_length, uint32_t spill_length) const // NOLINT(build/unsigned)
69{
70
71 // Enables the internal spill generator.
72 // Configures the internal command generator to produce spills at a defined frequency and length
73
74 // Rate = 50 Mhz / 2**( 12 + divider ) for divider between 0 and 15
75 // cyc_len and spill_len are in units of 1 / (50MHz / 2^24) = 0.34s
76
77 getNode("csr.ctrl.fake_cyc_len").write(cycle_length);
78 getNode("csr.ctrl.fake_spill_len").write(spill_length);
79 getNode("csr.ctrl.src").write(1);
80 getNode("csr.ctrl.en").write(1);
81 getClient().dispatch();
82 TLOG() << "Fake spills enabled";
83}
