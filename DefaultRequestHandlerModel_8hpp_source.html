63// This function takes the type returned by the begin() and end()
64// functions in a ReadoutType object and returns the timestamp that
65// should be used to determine whether the item pointed to by the
66// iterator is within the readout window. The "frame" type
67// ReadoutTypes all have a get_timestamp() function on the type
68// pointed to by the iterator (eg, WIBFrame), so we make this
69// "default" function return that. For other types, eg, those coming
70// from DS, there isn't a get_timestamp() function, so a different
71// template specialization is used in the appropriate package (eg,
72// trigger)
73template<class T>
74uint64_t
75get_frame_iterator_timestamp(T iter)
76{
77 return iter->get_timestamp();
78}

189 // An inline helper function that merges a set of byte arrays into a destination array
190 inline
191 void dump_to_buffer(const void* data, std::size_t size,
192 void* buffer, uint32_t buffer_pos, // NOLINT(build/unsigned)
193 const std::size_t& buffer_size)
194 {
195 auto bytes_to_copy = size;
196 while (bytes_to_copy > 0) {
197 auto n = std::min(bytes_to_copy, buffer_size - buffer_pos);
198 std::memcpy(static_cast<char*>(buffer) + buffer_pos, static_cast<const char*>(data), n);
199 buffer_pos += n;
200 bytes_to_copy -= n;
201 if (buffer_pos == buffer_size) {
202 buffer_pos = 0;
203 }
204 }
205 }
