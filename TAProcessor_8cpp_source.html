66void
67TAProcessor::conf(const appdal::ReadoutModule* conf)
68{
69 for (auto output : conf->get_outputs()) {
70 try {
71 if (output->get_data_type() == "TriggerCandidate") {
72 m_tc_sink = get_iom_sender<triggeralgs::TriggerCandidate>(output->UID());
73 }
74 } catch (const ers::Issue& excpt) {
75 ers::error(readoutlibs::ResourceQueueError(ERS_HERE, "tc", "DefaultRequestHandlerModel", excpt));
76 }
77 }
78
79 m_sourceid.id = conf->get_source_id();
80 m_sourceid.subsystem = trigger::TAWrapper::subsystem;
81 std::vector<const appdal::TCAlgorithm*> tc_algorithms;
82 auto dp = conf->get_module_configuration()->get_data_processor();
83 auto proc_conf = dp->cast<appdal::TADataProcessor>();
84 if (proc_conf != nullptr && proc_conf->get_mask_processing() == false ) {
85 tc_algorithms = proc_conf->get_algorithms();
86 }
87
88 for (auto algo : tc_algorithms) {
89 TLOG() << "Selected TC algorithm: " << algo->UID();
90 std::shared_ptr<triggeralgs::TriggerCandidateMaker> maker = make_tc_maker(algo->class_name());
91 nlohmann::json algo_json = algo->to_json(true);
92 maker->configure(algo_json[algo->UID()]);
93 inherited::add_postprocess_task(std::bind(&TAProcessor::find_tc, this, std::placeholders::_1, maker));
94 m_tcms.push_back(maker);
95 }
96 inherited::conf(conf);
97}

111void
112TAProcessor::find_tc(const TAWrapper* ta, std::shared_ptr<triggeralgs::TriggerCandidateMaker> tca)
113{
114 std::vector<triggeralgs::TriggerCandidate> tcs;
115 tca->operator()(ta->activity, tcs);
116 for (auto tc : tcs) {
117 if(!m_tc_sink->try_send(std::move(tc), iomanager::Sender::s_no_block)) {
118 ers::warning(TCDropped(ERS_HERE, tc.time_start, m_sourceid.id));
119 m_tcs_dropped++;
120 }
121 m_new_tcs++;
122 }
123 return;
124}
