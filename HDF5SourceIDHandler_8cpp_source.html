30hdf5rawdatafile::SrcIDGeoIDMap
31HDF5SourceIDHandler::rebuild_srcidgeoidmap(const source_id_geo_id_map_t& the_map) {
32
33 hdf5rawdatafile::SrcIDGeoIDMap m;
34 for( const auto& [sid, geoids] : the_map ) {
35 // There could be more than one, but we don't want to think about that
36 uint64_t geoid = *geoids.begin();
37 m.emplace_back(hdf5rawdatafile::SrcIDGeoIDEntry{
38 sid.id,
39 hdf5rawdatafile::GeoID{
40 .det_id = (geoid >> 0) & 0xff,
41 .crate_id = (geoid >> 16) & 0xff,
42 .slot_id = (geoid >> 32) & 0xff,
43 .stream_id = (geoid >> 48) & 0xff,
44 }
45 }
46 );
47 }
48
49 return m;
50}
