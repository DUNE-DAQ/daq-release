32class IntRange(click.types.IntParamType):
33 """A parameter that works similar to :data:`click.INT` but restricts
34 the value to fit into a range. The default behavior is to fail if the
35 value falls outside the range, but it can also be silently clamped
36 between the two edges.
37
38 See :ref:`ranges` for an example.
39 """
40 name = 'integer range'
41
42 def __init__(self, min=None, max=None, clamp=False):
43 self.min = min
44 self.max = max
45 self.clamp = clamp
46
47 def convert(self, value, param, ctx):
48
49 if type(value) == str:
50 if value.startswith('0x'):
51 base = 16
52 elif value.startswith('0o'):
53 bae = 8
54 elif value.startswith('0b'):
55 base = 2
56 else:
57 base = 10
58 rv = int(value, base)
59 else:
60 rv = int(value)
61 if self.clamp:
62 if self.min is not None and rv < self.min:
63 return self.min
64 if self.max is not None and rv > self.max:
65 return self.max
66 if self.min is not None and rv < self.min or \
67 self.max is not None and rv > self.max:
68 if self.min is None:
69 self.fail('%s is bigger than the maximum valid value '
70 '%s.' % (rv, self.max), param, ctx)
71 elif self.max is None:
72 self.fail('%s is smaller than the minimum valid value '
73 '%s.' % (rv, self.min), param, ctx)
74 else:
75 self.fail('%s is not in the valid range of %s to %s.'
76 % (rv, self.min, self.max), param, ctx)
77 return rv
78
79 def __repr__(self):
80 return 'IntRange(%r, %r)' % (self.min, self.max)

210def split_ints(ctx, param, value):
211
212 sep = ','
213 dash = '-'
214
215 if value is None:
216 return []
217
218 numbers = []
219 for item in value.split(sep):
220 nums = item.split(dash)
221 if len(nums) == 1:
222 # single entry
223 numbers.append(__str2int__(item))
224 elif len(nums) == 2:
225 # range
226 i, j = __str2int__(nums[0]), __str2int__(nums[1])
227 if i > j:
228 click.ClickException('Invalid interval '+item)
229 numbers.extend(list(range(i,j+1)))
230 else:
231 click.ClickException('Malformed option (comma separated list expected): {}'.format(value))
232
233 return numbers
