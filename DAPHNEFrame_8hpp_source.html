74 uint16_t get_adc(int i) const // NOLINT
75 {
76 if (i < 0 || i >= s_num_adcs)
77 throw std::out_of_range("ADC index out of range");
78
79 // The index of the first (and sometimes only) word containing the required ADC value
80 int word_index = s_bits_per_adc * i / s_bits_per_word;
81 assert(word_index < s_num_adc_words);
82 // Where in the word the lowest bit of our ADC value is located
83 int first_bit_position = (s_bits_per_adc * i) % s_bits_per_word;
84 // How many bits of our desired ADC are located in the `word_index`th word
85 int bits_from_first_word = std::min(s_bits_per_adc, s_bits_per_word - first_bit_position);
86 uint16_t adc = adc_words[word_index] >> first_bit_position; // NOLINT
87 // If we didn't get the full 14 bits from this word, we need the rest from the next word
88 if (bits_from_first_word < s_bits_per_adc) {
89 assert(word_index + 1 < s_num_adc_words);
90 adc |= adc_words[word_index + 1] << bits_from_first_word;
91 }
92 // Mask out all but the lowest 14 bits;
93 return adc & 0x3FFFu;
94 }
95
99 void set_adc(int i, uint16_t val) // NOLINT
100 {
101 if (i < 0 || i >= s_num_adcs)
102 throw std::out_of_range("ADC index out of range");
103 if (val >= (1 << s_bits_per_adc))
104 throw std::out_of_range("ADC value out of range");
105
106 // The index of the first (and sometimes only) word containing the required ADC value
107 int word_index = s_bits_per_adc * i / s_bits_per_word;
108 assert(word_index < s_num_adc_words);
109 // Where in the word the lowest bit of our ADC value is located
110 int first_bit_position = (s_bits_per_adc * i) % s_bits_per_word;
111 // How many bits of our desired ADC are located in the `word_index`th word
112 int bits_in_first_word = std::min(s_bits_per_adc, s_bits_per_word - first_bit_position);
113 uint32_t mask = (1 << (first_bit_position)) - 1;
114 adc_words[word_index] = ((val << first_bit_position) & ~mask) | (adc_words[word_index] & mask);
115 // If we didn't put the full 14 bits in this word, we need to put the rest in the next word
116 if (bits_in_first_word < s_bits_per_adc) {
117 assert(word_index + 1 < s_num_adc_words);
118 mask = (1 << (s_bits_per_adc - bits_in_first_word)) - 1;
119 adc_words[word_index + 1] = ((val >> bits_in_first_word) & mask) | (adc_words[word_index + 1] & ~mask);
120 }
121 }
