73template<class ReadoutType>
74void
75RecorderModel<ReadoutType>::do_work()
76{
77 m_time_point_last_info = std::chrono::steady_clock::now();
78
79 ReadoutType element;
80 while (m_run_marker) {
81 try {
82 element = m_data_receiver->receive(std::chrono::milliseconds(100)); // RS -> Use confed timeout?
83 m_packets_processed_total++;
84 m_packets_processed_since_last_info++;
85 if (!m_buffered_writer.write(reinterpret_cast<char*>(&element), sizeof(element))) { // NOLINT
86 ers::warning(CannotWriteToFile(ERS_HERE, m_output_file));
87 break;
88 }
89 } catch (const dunedaq::iomanager::TimeoutExpired& excpt) {
90 continue;
91 }
92 }
93 m_buffered_writer.flush();
94}
