77 tsNetDesc = rule->get_descriptor();
78 }
79 if (data_type == "DataRequest") {
80 dlhReqInputNetDesc = rule->get_descriptor();

84 hsiNetDesc = rule->get_descriptor();

95 if (dlhReqInputNetDesc == nullptr) {
96 throw(BadConf(ERS_HERE, "No DLH request input network descriptor given"));

99 throw(BadConf(ERS_HERE, "No HSIEvent output network descriptor given"));
100 }
101
102 auto idconf = get_source_id();

111 TLOG_DEBUG(7) << "creating OKS configuration object for Data Link Handler class " << dlhClass << ", id " << id;
112 confdb->create(dbfile, dlhClass, uid, dlhObj);
113 dlhObj.set_by_val<uint32_t>("source_id", id);

144 faNetObj.set_by_val<std::string>("data_type", dlhReqInputNetDesc->get_data_type());
145 faNetObj.set_obj("associated_service", &faServiceObj);

156 hsiNetObj.set_by_val<std::string>("data_type", hsiNetDesc->get_data_type());
157 hsiNetObj.set_obj("associated_service", &hsiServiceObj);
158
159 std::string genuid("FakeHSI-" + std::to_string(id));
160 conffwk::ConfigObject fakehsiObj;
161 confdb->create(dbfile, "FakeHSIEventGenerator", genuid, fakehsiObj);
162 fakehsiObj.set_obj("configuration", &rdrConf->config_object());
163 fakehsiObj.set_objs("outputs", { &queueObj, &hsiNetObj });
164
165 modules.push_back(confdb->get<FakeHSIEventGenerator>(genuid));
166
167 return modules;
