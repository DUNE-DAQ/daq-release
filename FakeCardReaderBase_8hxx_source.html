57void
58FakeCardReaderBase::do_conf(const nlohmann::json& /*args*/)
59{
60 TLOG_DEBUG(dunedaq::readoutlibs::logging::TLVL_ENTER_EXIT_METHODS) << get_fcr_name() << ": Entering do_conf() method";
61
62 if (m_configured) {
63 TLOG_DEBUG(dunedaq::readoutlibs::logging::TLVL_WORK_STEPS) << "This module is already configured!";
64 } else {
65 auto cfg = m_cfg->module<appdal::DataReader>(get_fcr_name());
66
67 std::map<uint32_t, const coredal::DROStreamConf*> streams;
68 for (const auto & readout_if : cfg->get_interfaces()) {
69 for (const auto& data_stream : readout_if->get_contains()) {
70 auto dro_stream = data_stream->cast<coredal::DROStreamConf>();
71 if (dro_stream != nullptr) {
72 streams[dro_stream->get_source_id()] = dro_stream;
73 }
74 }
75 }
76
77 for (const auto& qi : cfg->get_outputs()) {
78 auto q_with_id = qi->cast<coredal::QueueWithId>();
79 if (q_with_id == nullptr) {
80 throw readoutlibs::FailedFakeCardInitialization(ERS_HERE, get_fcr_name(), "Queue is not of type QueueWithId");
81 }
82 if (m_source_emus.find(q_with_id->UID()) == m_source_emus.end()) {
83 TLOG() << "Cannot find queue: " << q_with_id->UID() << std::endl;
84 throw readoutlibs::GenericConfigurationError(ERS_HERE, "Cannot find queue: " + q_with_id->UID());
85 }
86 if (m_source_emus[q_with_id->UID()]->is_configured()) {
87 TLOG() << "Emulator for queue name " << q_with_id->UID() << " was already configured";
88 throw readoutlibs::GenericConfigurationError(ERS_HERE, "Emulator configured twice: " + q_with_id->UID());
89 }
90 m_source_emus[q_with_id->UID()]->conf(streams[q_with_id->get_source_id()], cfg->get_configuration()->get_emulation_conf());
91 }
92 for (auto& [name, emu] : m_source_emus) {
93 if (!emu->is_configured()) {
94 throw readoutlibs::GenericConfigurationError(ERS_HERE, "Not all links were configured");
95 }
96 }
97
98 // Mark configured
99 m_configured = true;
100 }
101
102 TLOG_DEBUG(dunedaq::readoutlibs::logging::TLVL_ENTER_EXIT_METHODS) << get_fcr_name() << ": Exiting do_conf() method";
103}
