475void
476WIBEthFrameProcessor::process_swtpg_hits(uint16_t* primfind_it, dunedaq::daqdataformats::timestamp_t timestamp)
477{
478
479 constexpr int clocksPerTPCTick = types::DUNEWIBEthTypeAdapter::samples_tick_difference;
480
481 uint16_t chan[16], hit_end[16], hit_charge[16], hit_tover[16], hit_peak_time[16], hit_peak_adc[16], left[16]; // NOLINT(build/unsigned)
482 unsigned int nhits = 0;
483
484 while (*primfind_it != swtpg_wibeth::MAGIC) {
485 // First, get all of the register values (including those with no hit) into local variables
486 for (int i = 0; i < 16; ++i) {
487 chan[i] = *primfind_it++; // NOLINT(runtime/increment_decrement)
488 }
489 for (int i = 0; i < 16; ++i) {
490 hit_end[i] = *primfind_it++; // NOLINT(runtime/increment_decrement)
491 }
492 for (int i = 0; i < 16; ++i) {
493 hit_charge[i] = *primfind_it++; // NOLINT(runtime/increment_decrement)
494 // TLOG() << "hit_charge:" << hit_charge[i];
495 }
496 for (int i = 0; i < 16; ++i) {
497 // hit_tover[i] = static_cast<uint16_t>(*primfind_it++); // NOLINT(runtime/increment_decrement)
498 hit_tover[i] = *primfind_it++; // NOLINT(runtime/increment_decrement)
499 }
500 for (int i = 0; i < 16; ++i) {
501 hit_peak_adc[i] = *primfind_it++; // NOLINT(runtime/increment_decrement)
502 }
503 for (int i = 0; i < 16; ++i) {
504 hit_peak_time[i] = *primfind_it++; // NOLINT(runtime/increment_decrement)
505 }
506 for (int i = 0; i < 16; ++i) {
507 left[i] = *primfind_it++; // NOLINT(runtime/increment_decrement)
508 }
509
510 // Now that we have all the register values in local
511 // variables, loop over the register index (ie, channel) and
512 // find the channels which actually had a hit, as indicated by
513 // nonzero value of hit_charge
514 for (int i = 0; i < 16; ++i) {
515 // AAA: condition on the left hits makes sure to count hits
516 // correctly when they are spread across multiple channels
517 if (hit_charge[i] && left[i] == swtpg_wibeth::MAGIC
518 && chan[i] != swtpg_wibeth::MAGIC) {
519
520 uint64_t tp_t_begin = timestamp + clocksPerTPCTick * ((int64_t)hit_end[i] - (int64_t)hit_tover[i]);
521 uint64_t tp_t_peak = tp_t_begin + clocksPerTPCTick * hit_peak_time[i];
522
523 // This channel had a hit ending here, so we can create and output the hit here
524 const uint16_t offline_channel = m_register_channels[chan[i]];
525 if (m_channel_mask_set.find(offline_channel) == m_channel_mask_set.end()) {
526 // May be needed for TPSet:
527 // uint64_t tspan = clocksPerTPCTick * hit_tover[i]; // is/will be this needed?
528 //
529
530 // For quick n' dirty debugging: print out time/channel of hits.
531 // Can then make a text file suitable for numpy plotting with, eg:
532 //
533 // sed -n -e 's/.*Hit: \‍(.*\‍) \‍(.*\‍).*/\1 \2/p' log.txt > hits.txt
534 //
535
536 trigger::TriggerPrimitiveTypeAdapter tp;
537 tp.tp.time_start = tp_t_begin;
538 tp.tp.time_peak = tp_t_peak;
539 tp.tp.time_over_threshold = uint64_t((hit_tover[i]) * clocksPerTPCTick);
540 tp.tp.channel = offline_channel;
541 tp.tp.adc_integral = hit_charge[i];
