95void RestEndpoint::handleResponseCommand(const cmdobj_t& cmd, dunedaq::cmdlib::cmd::CommandReply& meta)
96{
97 dunedaq::cmdlib::cmd::Command command = cmd.get<dunedaq::cmdlib::cmd::Command>();
98 std::ostringstream addrstr;
99 addrstr << meta.data["ans-host"].get<std::string>() << ":" << meta.data["ans-port"].get<std::string>() << "/response";
100 meta.data["cmdid"] = command.id;
101 TLOG() << "Sending POST request to " << addrstr.str();
102
103 nlohmann::json body_json;
104 dunedaq::cmdlib::cmd::to_json(body_json, meta);
105 auto response = http_client_->post(addrstr.str()).body(body_json.dump()).send();
106 response.then(
107 [&](Http::Response response) {
108 TLOG() << "Response code = " << response.code();

110 [&](std::exception_ptr exc) {
111 // handle response failure

113 std::rethrow_exception(exc);
114 }
115 catch (const std::exception &e) {
116 TLOG() << "Exception thrown by Http::Client::post() call: \"" << e.what() << "\"; errno = " << errno;
117 }
118 }
119 );
120 http_client_responses_.push_back(std::move(response));
