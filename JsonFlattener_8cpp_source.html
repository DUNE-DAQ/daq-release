40JsonFlattener::parse_json(std::string path, const nlohmann::json& j)
41{
42
43 for (auto& [key, obj] : j.items()) {
44
45 auto objpath = (path.size() ? path + m_separator + key : key);
46
47 // validate meta filelds, i.e.
48 if (!obj.is_object())
49 throw OpmonJSONValidationError(ERS_HERE, key + " is not an object");
50
51 // if properties are here, process them
52 if (obj.count(JSONTags::properties)) {
53 // Loop over property structures
54 for (auto& pstruct : obj[JSONTags::properties].items()) {
55
56 // Make sure that this is a json object
57 if (!pstruct.value().is_object())
58 throw OpmonJSONValidationError(ERS_HERE, pstruct.key() + " is not an object");
59
60 // And that it contains the required fields
61 if (pstruct.value().count(JSONTags::time) == 0 || pstruct.value().count(JSONTags::data) == 0)
62 throw OpmonJSONValidationError(
63 ERS_HERE, pstruct.key() + " has no " + JSONTags::time + " or " + JSONTags::data + " tag");
64
65 // Check for presence of stubstructures
66 std::vector<std::string> sub_structs;
67 for (auto& pobj : pstruct.value().at(JSONTags::data).items()) {
68 if (pobj.value().is_object()) {
69 sub_structs.push_back(pobj.key());
70 }
71 }
72 if (sub_structs.size()) {
73 throw OpmonJSONValidationError(ERS_HERE, pstruct.key() + " contains substructures");
74 }
75
76 nlohmann::json entry;
77 auto& temp_tags = entry["tags"] = m_tags;
78 temp_tags["source_id"] = objpath;
79 entry["measurement"] = pstruct.key();
80 entry["fields"] = pstruct.value().at(JSONTags::data);
81 auto seconds = pstruct.value().at(JSONTags::time).get<std::time_t>();
82 auto gmt = gmtime(&seconds);
83 std::array<char, 80> time_c_string;
84 strftime(time_c_string.data(), 80, "%Y-%m-%dT%H:%M:%SZ", gmt);
85 std::string time_string(time_c_string.data());

87 m_components.push_back(entry);
88 }
89 }
