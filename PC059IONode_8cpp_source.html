53void
54PC059IONode::reset(int32_t fanout_mode, const std::string& clock_config_file) const
55{
56
57 // Soft reset
58 write_soft_reset_register();
59
60 millisleep(1000);
61
62 // Reset PLL and I2C
63 getNode("csr.ctrl.pll_rst").write(0x1);
64 getNode("csr.ctrl.pll_rst").write(0x0);
65
66 getNode("csr.ctrl.rst_i2c").write(0x1);
67 getNode("csr.ctrl.rst_i2c").write(0x0);
68
69 getNode("csr.ctrl.rst_i2cmux").write(0x1);
70 getNode("csr.ctrl.rst_i2cmux").write(0x0);
71
72 getClient().dispatch();
73
74 // enclustra i2c switch stuff
75 try {
76 getNode<I2CMasterNode>(m_uid_i2c_bus).get_slave("AX3_Switch").write_i2c(0x01, 0x7f);
77 } catch (const std::exception& e) {
78 ers::warning(EnclustraSwitchFailure(ERS_HERE, e));
79 }
80
81 // Find the right pll config file
82 std::string clock_config_path = get_full_clock_config_file_path(clock_config_file, fanout_mode);
83 TLOG() << "PLL configuration file : " << clock_config_path;
84
85 // Upload config file to PLL
86 configure_pll(clock_config_path);
87
88 // Reset mmcm
89 getNode("csr.ctrl.rst").write(0x1);
90 getNode("csr.ctrl.rst").write(0x0);
91 getClient().dispatch();
92
93 getNode("csr.ctrl.mux").write(0);
94 getClient().dispatch();
95
96 auto sfp_expander = get_i2c_device<I2CExpanderSlave>(m_uid_i2c_bus, "SFPExpander");
97
98 // Set invert registers to default for both banks
99 sfp_expander->set_inversion(0, 0x00);
100 sfp_expander->set_inversion(1, 0x00);
101
102 // Bank 0 input, bank 1 output
103 sfp_expander->set_io(0, 0x00);
104 sfp_expander->set_io(1, 0xff);
105
106 // Bank 0 - enable all SFPGs (enable low)
107 sfp_expander->set_outputs(0, 0x00);
108 TLOG_DEBUG(0) << "SFPs 0-7 enabled";
109
110 // To be removed from firmware address maps also
111 //getNode("csr.ctrl.rst_lock_mon").write(0x1);
112 //getNode("csr.ctrl.rst_lock_mon").write(0x0);
113 //getClient().dispatch();
114
115 TLOG() << "Reset done";
116}

166std::string
167PC059IONode::get_sfp_status(uint32_t sfp_id, bool print_out) const // NOLINT(build/unsigned)
168{
169 // on this board the upstream sfp has its own i2c bus, and the 8 downstream sfps are muxed onto the main i2c bus
170 std::stringstream status;
171 uint32_t sfp_bus_index; // NOLINT(build/unsigned)
172 if (sfp_id == 0) {
173 sfp_bus_index = 0;
174 status << "Upstream SFP:" << std::endl;
175 } else if (sfp_id > 0 && sfp_id < 9) {
176 switch_sfp_i2c_mux_channel(sfp_id - 1);
177 status << "Fanout SFP " << sfp_id - 1 << ":" << std::endl;
178 sfp_bus_index = 1;
179 } else {
180 throw InvalidSFPId(ERS_HERE, format_reg_value(sfp_id));
181 }
182 auto sfp = get_i2c_device<I2CSFPSlave>(m_sfp_i2c_buses.at(sfp_bus_index), "SFP_EEProm");
183
184 status << sfp->get_status();
185
186 if (print_out)
187 TLOG() << status.str();
188 return status.str();
189}

238void
239PC059IONode::get_info(opmonlib::InfoCollector& ci, int level) const
240{
241 if (level >= 2)
242 {
243 timinghardwareinfo::TimingPLLMonitorData pll_mon_data;
244 this->get_pll()->get_info(pll_mon_data);
245 ci.add(pll_mon_data);
246
247 timinghardwareinfo::TimingSFPMonitorData upstream_sfp_mon_data;
248 auto upstream_sfp = get_i2c_device<I2CSFPSlave>(m_sfp_i2c_buses.at(0), "SFP_EEProm");
249 try {
250 upstream_sfp->get_info(upstream_sfp_mon_data);
251 opmonlib::InfoCollector upstream_sfp_ic;
252 upstream_sfp_ic.add(upstream_sfp_mon_data);
253 ci.add("upstream_sfp", upstream_sfp_ic);
254 }
255 catch (timing::SFPUnreachable& e) {
256 // It is valid that an SFP may not be installed, currently no good way of knowing whether they it should be
257 TLOG_DEBUG(2) << "Failed to communicate with upstream SFP on i2c bus" << m_sfp_i2c_buses.at(0);
258 }
259
260
261 for (uint sfp_id=0; sfp_id < 8; ++sfp_id) {
262 TLOG_DEBUG(5) << "checking sfp: " << sfp_id;
263 switch_sfp_i2c_mux_channel(sfp_id);
264
265 auto sfp = get_i2c_device<I2CSFPSlave>(m_sfp_i2c_buses.at(1), "SFP_EEProm");
266 timinghardwareinfo::TimingSFPMonitorData sfp_data;
267
268 try {
269 sfp->get_info(sfp_data);
270 } catch (timing::SFPUnreachable& e) {
271 // It is valid that an SFP may not be installed, currently no good way of knowing whether they it should be
272 TLOG_DEBUG(2) << "Failed to communicate with downstream SFP: " << sfp_id << " on i2c bus" << m_sfp_i2c_buses.at(1);
273 continue;
274 }
275
276 opmonlib::InfoCollector sfp_ic;
277 sfp_ic.add(sfp_data);
278 ci.add("sfp_"+std::to_string(sfp_id), sfp_ic);
279 }
280 }
281
282 if (level >= 1) {
283 timinghardwareinfo::TimingPC059MonitorData mon_data;
284 this->get_info(mon_data);
285 ci.add(mon_data);
286 }
287}
