124template<class RDT, class LBT>
125void
126DefaultRequestHandlerModel<RDT, LBT>::record(const nlohmann::json& /*args*/)
127{
128 //auto conf = args.get<readoutconfig::RecordingParams>();
129 //FIXME: how do we pass the duration or recording?
130 int recording_time_sec = 1;
131 if (m_recording.load()) {
132 ers::error(CommandError(ERS_HERE, m_sourceid, "A recording is still running, no new recording was started!"));
133 return;
134 } else if (!m_buffered_writer.is_open()) {
135 ers::error(CommandError(ERS_HERE, m_sourceid, "DLH is not configured for recording"));
136 return;
137 }
138 m_recording_thread.set_work(
139 [&](int duration) {
140 TLOG() << "Start recording for " << duration << " second(s)" << std::endl;
141 m_recording.exchange(true);
142 auto start_of_recording = std::chrono::high_resolution_clock::now();
143 auto current_time = start_of_recording;
144 m_next_timestamp_to_record = 0;
145 RDT element_to_search;
146 while (std::chrono::duration_cast<std::chrono::seconds>(current_time - start_of_recording).count() < duration) {
147 if (!m_cleanup_requested || (m_next_timestamp_to_record == 0)) {
148 if (m_next_timestamp_to_record == 0) {
149 auto front = m_latency_buffer->front();
150 m_next_timestamp_to_record = front == nullptr ? 0 : front->get_first_timestamp();
151 }
152 element_to_search.set_first_timestamp(m_next_timestamp_to_record);
153 size_t processed_chunks_in_loop = 0;
154
155 {
156 std::unique_lock<std::mutex> lock(m_cv_mutex);
157 m_cv.wait(lock, [&] { return !m_cleanup_requested; });
158 m_requests_running++;
159 }
160 m_cv.notify_all();
161 auto chunk_iter = m_latency_buffer->lower_bound(element_to_search, true);
162 auto end = m_latency_buffer->end();
163 {
164 std::lock_guard<std::mutex> lock(m_cv_mutex);
165 m_requests_running--;
166 }
167 m_cv.notify_all();
168
169 for (; chunk_iter != end && chunk_iter.good() && processed_chunks_in_loop < 1000;) {
170 if ((*chunk_iter).get_first_timestamp() >= m_next_timestamp_to_record) {
171 if (!m_buffered_writer.write(reinterpret_cast<char*>(chunk_iter->begin()), // NOLINT
172 chunk_iter->get_payload_size())) {
173 ers::warning(CannotWriteToFile(ERS_HERE, m_output_file));
174 }
175 m_payloads_written++;
176 processed_chunks_in_loop++;
177 m_next_timestamp_to_record = (*chunk_iter).get_first_timestamp() +
178 RDT::expected_tick_difference * (*chunk_iter).get_num_frames();
179 }
180 ++chunk_iter;
181 }
182 }
183 current_time = std::chrono::high_resolution_clock::now();
184 }
185 m_next_timestamp_to_record = std::numeric_limits<uint64_t>::max(); // NOLINT (build/unsigned)
186
187 TLOG() << "Stop recording" << std::endl;
188 m_recording.exchange(false);
189 m_buffered_writer.flush();
190 },
191 recording_time_sec);
192}

207template<class RDT, class LBT>
208void
209DefaultRequestHandlerModel<RDT, LBT>::issue_request(dfmessages::DataRequest datarequest,
210 bool send_partial_fragment_if_available)
211{
212 boost::asio::post(*m_request_handler_thread_pool, [&, send_partial_fragment_if_available, datarequest]() { // start a thread from pool
213 auto t_req_begin = std::chrono::high_resolution_clock::now();
214 {
215 std::unique_lock<std::mutex> lock(m_cv_mutex);
216 m_cv.wait(lock, [&] { return !m_cleanup_requested; });
217 m_requests_running++;
218 }
219 m_cv.notify_all();
220 auto result = data_request(datarequest, send_partial_fragment_if_available);
221 {
222 std::lock_guard<std::mutex> lock(m_cv_mutex);
223 m_requests_running--;
224 }
225 m_cv.notify_all();
226 if (result.result_code == ResultCode::kFound || result.result_code == ResultCode::kNotFound) {
227 try { // Send to fragment connection
228 TLOG_DEBUG(TLVL_WORK_STEPS) << "Sending fragment with trigger/sequence_number "
229 << result.fragment->get_trigger_number() << "."
230 << result.fragment->get_sequence_number() << ", run number "
231 << result.fragment->get_run_number() << ", and SourceID "
232 << result.fragment->get_element_id() << ", and size "
233 << result.fragment->get_size() << ", and result code "
234 << result.result_code;
235 // Send fragment
236 get_iom_sender<std::unique_ptr<daqdataformats::Fragment>>(datarequest.data_destination)
237 ->send(std::move(result.fragment), std::chrono::milliseconds(m_fragment_send_timeout_ms));
238
239 } catch (const ers::Issue& excpt) {
240 ers::warning(CannotWriteToQueue(ERS_HERE, m_sourceid, datarequest.data_destination, excpt));
241 }
242 } else if (result.result_code == ResultCode::kNotYet) {
243 TLOG_DEBUG(TLVL_WORK_STEPS) << "Re-queue request. "
244 << "With timestamp=" << result.data_request.trigger_timestamp;
245 std::lock_guard<std::mutex> wait_lock_guard(m_waiting_requests_lock);
246 m_waiting_requests.push_back(RequestElement(datarequest, std::chrono::high_resolution_clock::now(),
247 send_partial_fragment_if_available));
248 }
249 auto t_req_end = std::chrono::high_resolution_clock::now();
250 auto us_req_took = std::chrono::duration_cast<std::chrono::microseconds>(t_req_end - t_req_begin);
251 TLOG_DEBUG(TLVL_WORK_STEPS) << "Responding to data request took: " << us_req_took.count() << "[us]";
252 // if (result.result_code == ResultCode::kFound) {
253 // std::lock_guard<std::mutex> time_lock_guard(m_response_time_log_lock);
254 // m_response_time_log.push_back( std::make_pair<int, int>(result.data_request.trigger_number,
255 // us_req_took.count()) );
256 // }
257 m_response_time_acc.fetch_add(us_req_took.count());
258 if ( us_req_took.count() > m_response_time_max.load() )
259 m_response_time_max.store(us_req_took.count());
260 if ( us_req_took.count() < m_response_time_min.load() )
261 m_response_time_min.store(us_req_took.count());
262 m_handled_requests++;
263 });
264}

354DefaultRequestHandlerModel<RDT, LBT>::cleanup()
355{
356 // auto now_s = time::now_as<std::chrono::seconds>();
357 auto size_guess = m_latency_buffer->occupancy();
358 if (size_guess > m_pop_limit_size) {

369 }
370 }

373 m_pops_count += popped;
374 m_error_registry->remove_errors_until(m_latency_buffer->front()->get_first_timestamp());
375 }
376 m_num_buffer_cleanups++;

379template<class RDT, class LBT>
380void
381DefaultRequestHandlerModel<RDT, LBT>::check_waiting_requests()
382{
383 // At run stop, we wait until all waiting requests have either:
384 //
385 // 1. been serviced because an item past the end of the window arrived in the buffer
386 // 2. timed out by going past m_request_timeout_ms, and returned a partial fragment
387 while (m_run_marker.load() || m_waiting_requests.size() > 0) {
388 {
389 std::lock_guard<std::mutex> lock_guard(m_waiting_requests_lock);
390
391 auto last_frame = m_latency_buffer->back(); // NOLINT
392 uint64_t newest_ts = last_frame == nullptr ? std::numeric_limits<uint64_t>::min() // NOLINT(build/unsigned)
393 : last_frame->get_first_timestamp();
394
395 size_t size = m_waiting_requests.size();
396
397 for (size_t i = 0; i < size;) {
398 if (m_waiting_requests[i].request.request_information.window_end < newest_ts) {
399 issue_request(m_waiting_requests[i].request, m_waiting_requests[i].send_partial_fragment_if_available);
400 std::swap(m_waiting_requests[i], m_waiting_requests.back());
401 m_waiting_requests.pop_back();
402 size--;
403 } else if (std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::high_resolution_clock::now() - m_waiting_requests[i].start_time).count() >= m_request_timeout_ms) {
404 issue_request(m_waiting_requests[i].request, true);
405
406 if (m_warn_on_timeout) {
407 ers::warning(dunedaq::readoutlibs::VerboseRequestTimedOut(ERS_HERE, m_sourceid,
408 m_waiting_requests[i].request.trigger_number,
409 m_waiting_requests[i].request.sequence_number,
410 m_waiting_requests[i].request.run_number,
411 m_waiting_requests[i].request.request_information.window_begin,
412 m_waiting_requests[i].request.request_information.window_end,
413 m_waiting_requests[i].request.data_destination));
414 }
415
416 m_num_requests_bad++;
417 m_num_requests_timed_out++;
418
419 std::swap(m_waiting_requests[i], m_waiting_requests.back());
420 m_waiting_requests.pop_back();
421 size--;
422 } else {
423 i++;
424 }
425 }
426 }
427 std::this_thread::sleep_for(std::chrono::milliseconds(10));
428 }
429}

431template<class RDT, class LBT>
432std::vector<std::pair<void*, size_t>>
433DefaultRequestHandlerModel<RDT, LBT>::get_fragment_pieces(uint64_t start_win_ts,
434 uint64_t end_win_ts,
435 RequestResult& rres)
436{
437
438 TLOG_DEBUG(TLVL_WORK_STEPS) << "Looking for frags between " << start_win_ts << " and " << end_win_ts;
439 std::vector<std::pair<void*, size_t>> frag_pieces;
440 RDT request_element = RDT();
441 request_element.set_first_timestamp(start_win_ts-(request_element.get_num_frames() * RDT::expected_tick_difference));
442 auto start_iter = m_error_registry->has_error("MISSING_FRAMES")
443 ? m_latency_buffer->lower_bound(request_element, true)
444 : m_latency_buffer->lower_bound(request_element, false);
445 if (start_iter == m_latency_buffer->end()) {
446 // Due to some concurrent access, the start_iter could not be retrieved successfully, try again
447 ++m_num_requests_delayed;
448 rres.result_code = ResultCode::kNotYet; // give it another chance
449 TLOG() << "Timestamp in future";
450 } else {
451 TLOG_DEBUG(TLVL_WORK_STEPS) << "Lower bound found " << start_iter->get_first_timestamp() << ", --> distance from window: " << int64_t(start_win_ts) - int64_t(start_iter->get_first_timestamp()) ;
452 rres.result_code = ResultCode::kFound;
453 ++m_num_requests_found;
454
455 auto elements_handled = 0;
456
457 RDT* element = &(*start_iter);
458
459 while (start_iter.good() && element->get_first_timestamp() < end_win_ts) {
460 //if ( element->get_first_timestamp() + (element->get_num_frames() - 1) * RDT::expected_tick_difference < start_win_ts) {
461 if ( element->get_first_timestamp() + element->get_num_frames() * RDT::expected_tick_difference <= start_win_ts) {
462 //TLOG() << "skip processing for current element " << element->get_first_timestamp() << ", out of readout window.";
463 }
464
465 else if ( element->get_num_frames()>1 &&
466 ((element->get_first_timestamp() < start_win_ts &&
467 element->get_first_timestamp() + element->get_num_frames() * RDT::expected_tick_difference > start_win_ts)
468 ||
469 element->get_first_timestamp() + element->get_num_frames() * RDT::expected_tick_difference >
470 end_win_ts)) {
471 //TLOG() << "We don't need the whole aggregated object (e.g.: superchunk)" ;
472 for (auto frame_iter = element->begin(); frame_iter != element->end(); frame_iter++) {
473 if (get_frame_iterator_timestamp(frame_iter) > (start_win_ts - RDT::expected_tick_difference)&&
474 get_frame_iterator_timestamp(frame_iter) < end_win_ts ) {
475 frag_pieces.emplace_back(
476 std::make_pair<void*, size_t>(static_cast<void*>(&(*frame_iter)), element->get_frame_size()));
477 }
478 }
479 }
480 else {
481 //TLOG() << "Add element " << element->get_first_timestamp();
482 // We are somewhere in the middle -> the whole aggregated object (e.g.: superchunk) can be copied
483 frag_pieces.emplace_back(
484 std::make_pair<void*, size_t>(static_cast<void*>((*start_iter).begin()), element->get_payload_size()));
485 }
486
487 elements_handled++;
488 ++start_iter;
489 element = &(*start_iter);
490 }
491
492 }
493 TLOG_DEBUG(TLVL_WORK_STEPS) << "*** Number of frames retrieved: " << frag_pieces.size();
494 return frag_pieces;
495}

497template<class RDT, class LBT>
498typename DefaultRequestHandlerModel<RDT, LBT>::RequestResult
499DefaultRequestHandlerModel<RDT, LBT>::data_request(dfmessages::DataRequest dr,
500 bool send_partial_fragment_if_available)
501{
502 // Prepare response
503 RequestResult rres(ResultCode::kUnknown, dr);
504
505 // Prepare FragmentHeader and empty Fragment pieces list
506 auto frag_header = create_fragment_header(dr);
507 std::vector<std::pair<void*, size_t>> frag_pieces;
508 std::ostringstream oss;
509
510 bool local_data_not_found_flag = false;
511 if (m_latency_buffer->occupancy() != 0) {
512 // Data availability is calculated here
513 auto front_element = m_latency_buffer->front(); // NOLINT
514 auto last_element = m_latency_buffer->back(); // NOLINT
515 uint64_t last_ts = front_element->get_first_timestamp(); // NOLINT(build/unsigned)
516 uint64_t newest_ts = last_element->get_first_timestamp(); // NOLINT(build/unsigned)
517
518 uint64_t start_win_ts = dr.request_information.window_begin; // NOLINT(build/unsigned)
519 uint64_t end_win_ts = dr.request_information.window_end; // NOLINT(build/unsigned)
520 TLOG_DEBUG(TLVL_WORK_STEPS) << "Data request for trig/seq_num=" << dr.trigger_number
521 << "." << dr.sequence_number << " and SourceID[" << m_sourceid << "] with"
522 << " Trigger TS=" << dr.trigger_timestamp
523 << " Oldest stored TS=" << last_ts
524 << " Newest stored TS=" << newest_ts
525 << " Start of window TS=" << start_win_ts
526 << " End of window TS=" << end_win_ts
527 << " Latency buffer occupancy=" << m_latency_buffer->occupancy();
528
529 // List of safe-extraction conditions
530 if (last_ts <= start_win_ts && end_win_ts <= newest_ts) { // the full window of data is there
531 frag_pieces = get_fragment_pieces(start_win_ts, end_win_ts, rres);
532 } else if (send_partial_fragment_if_available && last_ts <= end_win_ts && end_win_ts <= newest_ts) { // partial data is there
533 frag_pieces = get_fragment_pieces(start_win_ts, end_win_ts, rres);
534 if (rres.result_code == ResultCode::kNotYet) {
535 // 15-Sep-2022, KAB: this is really ugly. I'm not sure why get_fragment_pieces occasionally
536 // returns kNotYet when running with long readout windows, but I suspect that it has something
537 // to do with that code assuming that the readout window is fully contained within the data
538 // that exists in the latency buffer. In any case, this code simply bails out when that happens.
539 local_data_not_found_flag = true;
540 rres.result_code = ResultCode::kNotFound;
541 }
542 } else if ((! send_partial_fragment_if_available) && last_ts > start_win_ts) { // data at the start of the window is missing
543 frag_header.error_bits |= (0x1 << static_cast<size_t>(daqdataformats::FragmentErrorBits::kDataNotFound));
544 rres.result_code = ResultCode::kNotFound;
545 ++m_num_requests_old_window;
546 ++m_num_requests_bad;
547 } else if (send_partial_fragment_if_available && last_ts > end_win_ts) { // data is completely gone
548 local_data_not_found_flag = true;
549 frag_header.error_bits |= (0x1 << static_cast<size_t>(daqdataformats::FragmentErrorBits::kDataNotFound));
550 rres.result_code = ResultCode::kNotFound;
551 ++m_num_requests_old_window;
552 ++m_num_requests_bad;
553 } else if (newest_ts < end_win_ts) { // data at the end of the window is missing (more could still arrive)
554 if (send_partial_fragment_if_available) {
555 // We've been asked to send the partial fragment if we don't
556 // have an object past the end of the window, so fill the
557 // fragment with what we have so far
558 if (m_warn_on_timeout) {
559 TLOG()
560 << "Returning partial fragment for trig/seq number " << dr.trigger_number << "." << dr.sequence_number
561 << " with TS " << dr.trigger_timestamp
562 << ". Component " << dr.request_information.component
563 << " with type " << daqdataformats::fragment_type_to_string(daqdataformats::FragmentType(frag_header.fragment_type));
564 } else {
565 TLOG_DEBUG(TLVL_WORK_STEPS)
566 << "Returning partial fragment for trig/seq number " << dr.trigger_number << "." << dr.sequence_number
567 << " with TS " << dr.trigger_timestamp
568 << ". Component " << dr.request_information.component
569 << " with type " << daqdataformats::fragment_type_to_string(daqdataformats::FragmentType(frag_header.fragment_type));
570 }
571 frag_header.error_bits |= (0x1 << static_cast<size_t>(daqdataformats::FragmentErrorBits::kIncomplete));
572 frag_pieces = get_fragment_pieces(start_win_ts, end_win_ts, rres);
573 // 06-Jul-2022, KAB: added the following line to translate a kNotYet status code from
574 // get_fragment_pieces() to kFound. The reasoning behind this addition is that when
575 // send_partial_fragment_if_available is set to true, we should accept whatever we've got
576 // in the buffer and *not* retry any longer. So, we force that behavior with the
577 // following line. The data-taking scenario which illustrates
578 // this situation is long-window-readout in which a given trigger is split into a sequence
579 // of TriggerRecords and the 2..Nth DataRequest find nothing useful in the TriggerCandidate
580 // (TCBuffer) latency buffer (and no data later than the request window end time).
581 // In those cases, get_fragment_pieces() returns kNotYet, and we need to change that
582 // kFound in order to not keep looping.
583 if (rres.result_code == ResultCode::kNotYet) {rres.result_code = ResultCode::kFound;}
584 } else {
585 ++m_num_requests_delayed;
586 rres.result_code = ResultCode::kNotYet; // give it another chance
587 }
588 } else {
589 TLOG() << "Don't know how to categorise this request";
590 frag_header.error_bits |= (0x1 << static_cast<size_t>(daqdataformats::FragmentErrorBits::kDataNotFound));
591 rres.result_code = ResultCode::kNotFound;
592 ++m_num_requests_uncategorized;
593 ++m_num_requests_bad;
594 }
595
596 // Requeue if needed
597 if (rres.result_code == ResultCode::kNotYet) {
598 if (m_run_marker.load()) {
599 return rres; // If kNotYet, return immediately, don't check for fragment pieces.
600 } else {
601 frag_header.error_bits |= (0x1 << static_cast<size_t>(daqdataformats::FragmentErrorBits::kDataNotFound));
602 rres.result_code = ResultCode::kNotFound;
603 ++m_num_requests_bad;
604 }
605 }
606
607 // Build fragment
608 oss << "TS match result for SourceID[" << m_sourceid << "]: "
609 << " Trigger/sequence number=" << dr.trigger_number << "." << dr.sequence_number
610 << " Oldest stored TS=" << last_ts
611 << " Start of window TS=" << start_win_ts
612 << " End of window TS=" << end_win_ts
613 << " Estimated newest stored TS=" << newest_ts
614 << " Requestor=" << dr.data_destination;
615 TLOG_DEBUG(TLVL_WORK_STEPS) << oss.str();
616 } else {
617 local_data_not_found_flag = true;
618 if (m_warn_about_empty_buffer) {
619 ers::warning(RequestOnEmptyBuffer(ERS_HERE, m_sourceid, "Data not found"));
620 } else {
621 TLOG_DEBUG(TLVL_WORK_STEPS) << "SourceID[" << m_sourceid << "] Request on empty buffer: Data not found";
622 }
623 frag_header.error_bits |= (0x1 << static_cast<size_t>(daqdataformats::FragmentErrorBits::kDataNotFound));
624 rres.result_code = ResultCode::kNotFound;
625 ++m_num_requests_bad;
626 }
627
628 if (rres.result_code != ResultCode::kFound) {
629 if (m_warn_about_empty_buffer || (! local_data_not_found_flag)) {
630 ers::warning(dunedaq::readoutlibs::TrmWithEmptyFragment(ERS_HERE, m_sourceid, oss.str()));
631 } else {
632 TLOG_DEBUG(TLVL_WORK_STEPS) << "SourceID[" << m_sourceid << "] Trigger Matching result with empty fragment: " << oss.str();
633 }
634 }
635
636 // Create fragment from pieces
637 rres.fragment = std::make_unique<daqdataformats::Fragment>(frag_pieces);
638
639 // Set header
640 rres.fragment->set_header_fields(frag_header);
641
642 return rres;
643}
