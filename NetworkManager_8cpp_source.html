55void
56NetworkManager::configure(const Connections_t& connections,
57 bool use_config_client,
58 std::chrono::milliseconds config_client_interval)
59{
60 if (!m_preconfigured_connections.empty()) {
61 throw AlreadyConfigured(ERS_HERE);
62 }
63
64 for (auto& connection : connections) {
65 auto name = connection.id.uid;
66 TLOG_DEBUG(15) << "Adding connection " << name << " to connection map";
67 if (m_preconfigured_connections.count(connection.id)) {
68 TLOG_DEBUG(15) << "Name collision for connection " << name << ", DT " << connection.id.data_type
69 << " connection_map.count: " << m_preconfigured_connections.count(connection.id);
70 reset();
71 throw NameCollision(ERS_HERE, connection.id.uid);
72 }
73 m_preconfigured_connections[connection.id] = connection;
74 }
75
76 if (use_config_client && m_config_client == nullptr) {
77
78 std::string connectionServer = "localhost";
79 char* env = getenv("CONNECTION_SERVER");
80 if (env) {
81 connectionServer = std::string(env);
82 }
83 std::string connectionPort = "5000";
84 env = getenv("CONNECTION_PORT");
85 if (env) {
86 connectionPort = std::string(env);
87 }
88 TLOG_DEBUG(17) << "ConnectionServer host and port are " << connectionServer << ":" << connectionPort;
89 m_config_client = std::make_unique<ConfigClient>(connectionServer, connectionPort, config_client_interval);
90 }
91 m_config_client_interval = config_client_interval;
92}

97 m_subscriber_update_thread_running = false;

99 m_subscriber_update_thread->join();

185ConnectionResponse
186NetworkManager::get_connections(ConnectionId const& conn_id, bool restrict_single) const
187{
188 auto response = get_preconfigured_connections(conn_id);
189 if (restrict_single && response.connections.size() > 1) {
190 throw NameCollision(ERS_HERE, conn_id.uid);
191 }
192 if (m_config_client != nullptr) {
193 auto start_time = std::chrono::steady_clock::now();
194 while (
195 std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::steady_clock::now() - start_time).count() <
196 1000) {
197 try {
198 auto client_response = m_config_client->resolveConnection(conn_id, conn_id.session);
199 if (restrict_single && client_response.connections.size() > 1) {
200 throw NameCollision(ERS_HERE, conn_id.uid);
201 }
202
203 if (client_response.connections.size() > 0) {
204 response = client_response;
205 }
206 break;
207 } catch (FailedLookup const& lf) {
208 if (m_config_client->is_connected()) {
209 throw ConnectionNotFound(ERS_HERE, conn_id.uid, conn_id.data_type, lf);
210 }
211 usleep(1000);
212 }
213 }
214 }
215
216 if (response.connections.size() == 0) {
217 throw ConnectionNotFound(ERS_HERE, conn_id.uid, conn_id.data_type);
218 }
219
220 return response;
221}

248std::shared_ptr<ipm::Receiver>
249NetworkManager::create_receiver(std::vector<ConnectionInfo> connections, ConnectionId const& conn_id)
250{
251 TLOG_DEBUG(12) << "START";
252 if (connections.size() == 0) {
253 return nullptr;
254 }
255
256 bool is_pubsub = connections[0].connection_type == ConnectionType::kPubSub;
257 if (connections.size() > 1 && !is_pubsub) {
258 throw OperationFailed(ERS_HERE,
259 "Trying to configure a kSendRecv receiver with multiple Connections is not allowed!");
260 }
261
262 auto plugin_type =
263 ipm::get_recommended_plugin_name(is_pubsub ? ipm::IpmPluginType::Subscriber : ipm::IpmPluginType::Receiver);
264
265 TLOG_DEBUG(12) << "Creating plugin of type " << plugin_type;
266 auto plugin = dunedaq::ipm::make_ipm_receiver(plugin_type);
267
268 nlohmann::json config_json;
269 if (is_pubsub) {
270 std::vector<std::string> uris;
271 for (auto& conn : connections) {
272 // Check for case where both ends are in app and ConnectivityService hasn't received other end yet
273 if (conn.uri.find("*") != std::string::npos || conn.uri.find("0.0.0.0") != std::string::npos) {
274 continue;
275 }
276 uris.push_back(conn.uri);
277 }
278 if (uris.size() == 0) {
279 return nullptr;
280 }
281 config_json["connection_strings"] = uris;
282 } else {
283 config_json["connection_string"] = connections[0].uri;
284 }
285 auto newCs = plugin->connect_for_receives(config_json);
286 TLOG_DEBUG(12) << "Receiver reports connected to URI " << newCs;
287
288 // Replace with resolved if there are wildcards (host and/or port)
289 if (connections[0].uri.find("*") != std::string::npos || connections[0].uri.find("0.0.0.0") != std::string::npos) {
290 TLOG_DEBUG(14) << "Wildcard found in connection URI " << connections[0].uri << ", adjusting before publish";
291 auto newUri = utilities::parse_connection_string(newCs);
292 auto oldUri = utilities::parse_connection_string(connections[0].uri);
293
294 if (oldUri.port == "*")
295 oldUri.port = newUri.port;
296 if (oldUri.host == "*" || oldUri.host == "0.0.0.0")
297 oldUri.host = newUri.host;
298
299 connections[0].uri = oldUri.to_string();
300 TLOG_DEBUG(14) << "Connection URI is now " << connections[0].uri;
301 }
302
303 if (is_pubsub) {
304 TLOG_DEBUG(12) << "Subscribing to topic " << connections[0].data_type << " after connect_for_receives";
305 auto subscriber = std::dynamic_pointer_cast<ipm::Subscriber>(plugin);
306 subscriber->subscribe(connections[0].data_type);
307 std::lock_guard<std::mutex> lkk(m_subscriber_plugin_map_mutex);
308 m_subscriber_plugins[conn_id] = subscriber;
309 if (!m_subscriber_update_thread_running) {
310 m_subscriber_update_thread_running = true;
311 m_subscriber_update_thread.reset(new std::thread(&NetworkManager::update_subscribers, this));
312 }
313 }
314
315 if (m_config_client != nullptr && !is_pubsub) {
316 m_config_client->publish(connections[0]);
317 }
318
319 TLOG_DEBUG(12) << "END";
320 return plugin;
321}

323std::shared_ptr<ipm::Sender>
324NetworkManager::create_sender(ConnectionInfo connection)
325{
326 auto is_pubsub = connection.connection_type == ConnectionType::kPubSub;
327 auto plugin_type =
328 ipm::get_recommended_plugin_name(is_pubsub ? ipm::IpmPluginType::Publisher : ipm::IpmPluginType::Sender);
329
330 // Check for case where both ends are in app and ConnectivityService hasn't received other end yet
331 if (!is_pubsub && (connection.uri.find("*") != std::string::npos || connection.uri.find("0.0.0.0") != std::string::npos)) {
332 return nullptr;
333 }
334
335 TLOG_DEBUG(11) << "Creating sender plugin of type " << plugin_type;
336 auto plugin = dunedaq::ipm::make_ipm_sender(plugin_type);
337 TLOG_DEBUG(11) << "Connecting sender plugin to " << connection.uri;
338 auto newCs = plugin->connect_for_sends({ { "connection_string", connection.uri } });
339 TLOG_DEBUG(11) << "Sender Plugin connected, reports URI " << newCs;
340
341 // Replace with resolved if there are wildcards (host and/or port)
342 if (connection.uri.find("*") != std::string::npos || connection.uri.find("0.0.0.0") != std::string::npos) {
343 TLOG_DEBUG(13) << "Wildcard found in connection URI " << connection.uri << ", adjusting before publish";
344 auto newUri = utilities::parse_connection_string(newCs);
345 auto oldUri = utilities::parse_connection_string(connection.uri);
346
347 if (oldUri.port == "*")
348 oldUri.port = newUri.port;
349 if (oldUri.host == "*" || oldUri.host == "0.0.0.0")
350 oldUri.host = newUri.host;
351
352 connection.uri = oldUri.to_string();
353 TLOG_DEBUG(13) << "Connection URI is now " << connection.uri;
354 }

356 if (m_config_client != nullptr && is_pubsub) {

358 }

363void
364NetworkManager::update_subscribers()
365{
366 while (m_subscriber_update_thread_running.load()) {
367 {
368 std::lock_guard<std::mutex> lk(m_subscriber_plugin_map_mutex);
369 for (auto& subscriber_pair : m_subscriber_plugins) {
370 try {
371 auto response = get_connections(subscriber_pair.first, false);
372
373 nlohmann::json config_json;
374 std::vector<std::string> uris;
375 for (auto& conn : response.connections)
376 uris.push_back(conn.uri);
377 config_json["connection_strings"] = uris;
378
379 subscriber_pair.second->connect_for_receives(config_json);
380 } catch (ers::Issue&) {
381 }
382 }
383 }
384 std::this_thread::sleep_for(m_config_client_interval);
385 }
386}
