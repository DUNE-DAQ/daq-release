59void
60FMCIONode::reset(const std::string& clock_config_file) const
61{
62
63 write_soft_reset_register();
64
65 millisleep(1000);
66
67 // Reset PLL
68 getNode("csr.ctrl.pll_rst").write(0x1);
69 getNode("csr.ctrl.pll_rst").write(0x0);
70 getClient().dispatch();
71
72 CarrierType carrier_type = convert_value_to_carrier_type(read_carrier_type());
73
74 // enclustra i2c switch stuff
75 if (carrier_type == kCarrierEnclustraA35) {
76 try {
77 getNode<I2CMasterNode>(m_uid_i2c_bus).get_slave("AX3_Switch").write_i2c(0x01, 0x7f);
78 } catch (const std::exception& e) {
79 ers::warning(EnclustraSwitchFailure(ERS_HERE, e));
80 }
81 }
82
83 // Find the right pll config file
84 std::string clock_config_path = get_full_clock_config_file_path(clock_config_file);
85 TLOG() << "PLL configuration file : " << clock_config_path;
86
87 // Upload config file to PLL
88 configure_pll(clock_config_path);
89
90 // Reset mmcm
91 getNode("csr.ctrl.rst").write(0x1);
92 getNode("csr.ctrl.rst").write(0x0);
93 getClient().dispatch();
94
95 // Enable sfp tx laser
96 getNode("csr.ctrl.sfp_tx_dis").write(0x0);
97
98 // rx edges
99 uint32_t cdr_rx_edge = 0x0; // NOLINT(build/unsigned)
100 uint32_t sfp_rx_edge = 0x0; // NOLINT(build/unsigned)
101 uint32_t rj45_rx_edge = 0x0; // NOLINT(build/unsigned)
102
103 // tx edges
104 uint32_t sfp_tx_edge = 0x0; // NOLINT(build/unsigned)
105 uint32_t rj45_tx_edge = 0x0; // NOLINT(build/unsigned)
106
107 // rx edges
108 getNode("csr.ctrl.cdr_rx_edge").write(cdr_rx_edge);
109 getNode("csr.ctrl.sfp_rx_edge").write(sfp_rx_edge);
110 getNode("csr.ctrl.rj45_rx_edge").write(rj45_rx_edge);
111
112 // tx edges
113 getNode("csr.ctrl.sfp_tx_edge").write(sfp_tx_edge);
114 getNode("csr.ctrl.rj45_tx_edge").write(rj45_tx_edge);
115
116 getClient().dispatch();
117
118 TLOG() << "Reset done";
119}

189void
190FMCIONode::get_info(opmonlib::InfoCollector& ci, int level) const
191{
192 if (level >= 2) {
193 timinghardwareinfo::TimingPLLMonitorData pll_mon_data;
194 this->get_pll()->get_info(pll_mon_data);
195 ci.add(pll_mon_data);
196
197 timinghardwareinfo::TimingSFPMonitorData sfp_mon_data;
198 auto sfp = this->get_i2c_device<I2CSFPSlave>(m_sfp_i2c_buses.at(0), "SFP_EEProm");
199 try {
200 sfp->get_info(sfp_mon_data);
201 ci.add(sfp_mon_data);
202 } catch (timing::SFPUnreachable& e) {
203 // It is valid that an SFP may not be installed, currently no good way of knowing whether they it should be
204 TLOG_DEBUG(2) << "Failed to communicate with SFP on i2c bus" << m_sfp_i2c_buses.at(0);
205 }
206 }
207 if (level >= 1) {
208 timinghardwareinfo::TimingFMCMonitorData mon_data;
209 this->get_info(mon_data);
210 ci.add(mon_data);
211 }
212}
