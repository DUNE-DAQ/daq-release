41TimestampEstimator::add_timestamp_datapoint(uint64_t daq_time, uint64_t system_time)
42{
43 std::scoped_lock<std::mutex> lk(m_datapoint_mutex);
44
45 // First, update the latest timestamp
46 uint64_t estimate = m_current_timestamp_estimate.load();
47 int64_t diff = estimate - daq_time;
48 TLOG_DEBUG(TLVL_TIME_SYNC_PROPERTIES) << "Got a TimeSync timestamp = " << daq_time
49 << ", system time = " << system_time
50 << " when current timestamp estimate was " << estimate << ". diff=" << diff;
51
52 if (m_most_recent_daq_time == std::numeric_limits<uint64_t>::max() ||
53 daq_time > m_most_recent_daq_time) {
54 m_most_recent_daq_time = daq_time;
55 m_most_recent_system_time = system_time;
56 }
57
58 if (m_most_recent_daq_time != std::numeric_limits<uint64_t>::max()) {
59 // Update the current timestamp estimate, based on the most recently-read TimeSync
60 using namespace std::chrono;
61
62 auto time_now =
63 static_cast<uint64_t>(duration_cast<microseconds>(system_clock::now().time_since_epoch()).count()); // NOLINT
64
65 // (PAR 2021-07-22) We only want to _increase_ our timestamp
66 // estimate, not _decrease_ it, so we only attempt the update if
67 // our system time is later than the latest time sync's system
68 // time. We can get TimeSync messages from the "future" if
69 // they're coming from another host whose clock is not exactly
70 // synchronized with ours: that's fine, but if the discrepancy
71 // is large, then badness could happen, so emit a warning
72
73 if (time_now < m_most_recent_system_time - 10000) {
74 ers::warning(EarlyTimeSync(ERS_HERE, m_most_recent_system_time - time_now));
75 }
76
77 if (time_now > m_most_recent_system_time) {
78
79 auto delta_time = time_now - m_most_recent_system_time;
80 TLOG_DEBUG(TLVL_TIME_SYNC_PROPERTIES)
81 << "Time diff between current system and latest TimeSync system time [us]: " << delta_time;
82
83 // Warn user if current system time is more than 1s ahead of latest TimeSync system time. This could be a sign of
84 // an issue, e.g. machine times out of sync
85 if (delta_time > 1e6)
86 ers::warning(LateTimeSync(ERS_HERE, delta_time));
87
88 const uint64_t new_timestamp =
89 m_most_recent_daq_time + delta_time * m_clock_frequency_hz / 1000000;
90
91 // Don't ever decrease the timestamp; just wait until enough
92 // time passes that we want to increase it
93 if (m_current_timestamp_estimate.load() == std::numeric_limits<uint64_t>::max() ||
94 new_timestamp >= m_current_timestamp_estimate.load()) {
95 TLOG_DEBUG(TLVL_TIME_SYNC_NEW_ESTIMATE)
96 << "Storing new timestamp estimate of " << new_timestamp << " ticks (..." << std::fixed
97 << std::setprecision(8)
98 << (static_cast<double>(new_timestamp % (m_clock_frequency_hz * 1000)) /
99 static_cast<double>(m_clock_frequency_hz))
100 << " sec), mrt.daq_time is " << m_most_recent_daq_time << " ticks (..."
101 << (static_cast<double>(m_most_recent_daq_time % (m_clock_frequency_hz * 1000)) /
102 static_cast<double>(m_clock_frequency_hz))
103 << " sec), delta_time is " << delta_time << " usec, clock_freq is " << m_clock_frequency_hz << " Hz";
104 m_current_timestamp_estimate.store(new_timestamp);
105 } else {
106 TLOG_DEBUG(TLVL_TIME_SYNC_NOTES) << "Not updating timestamp estimate backwards from "
107 << m_current_timestamp_estimate.load() << " to " << new_timestamp;
108 }
109 }
110 }
