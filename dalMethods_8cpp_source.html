47static void
48make_parents_list(
49 const ConfigObjectImpl * child,
50 const dunedaq::coredal::ResourceSet * resource_set,
51 std::vector<const dunedaq::coredal::Component *> & p_list,
52 std::list< std::vector<const dunedaq::coredal::Component *> >& out,
53 dunedaq::coredal::TestCircularDependency& cd_fuse)
54{
55 dunedaq::coredal::AddTestOnCircularDependency add_fuse_test(cd_fuse, resource_set);
56
57 // add the resource set to the path
58 p_list.push_back(resource_set);
59
60 // check if the application is in the resource relationship, i.e. is a resource or belongs to resource set(s)
61 for (const auto& i : resource_set->get_contains()) {
62 if (i->config_object().implementation() == child) {
63 out.push_back(p_list);
64 }
65 else if (const dunedaq::coredal::ResourceSet * rs = i->cast<dunedaq::coredal::ResourceSet>()) {
66 make_parents_list(child, rs, p_list, out, cd_fuse);
67 }
68 }
69
70 // remove the resource set from the path
71 p_list.pop_back();
72}

74static void
75make_parents_list(
76 const ConfigObjectImpl * child,
77 const dunedaq::coredal::Segment * segment,
78 std::vector<const dunedaq::coredal::Component *> & p_list,
79 std::list<std::vector<const dunedaq::coredal::Component *> >& out,
80 bool is_segment,
81 dunedaq::coredal::TestCircularDependency& cd_fuse)
82{
83 dunedaq::coredal::AddTestOnCircularDependency add_fuse_test(cd_fuse, segment);
84
85 // add the segment to the path
86 p_list.push_back(segment);
87
88 // check if the application is in the nested segment
89 for (const auto& seg : segment->get_segments()) {
90 if (seg->config_object().implementation() == child)
91 out.push_back(p_list);
92 else
93 make_parents_list(child, seg, p_list, out, is_segment, cd_fuse);
94 }
95 if (!is_segment) {
96 for (const auto& app : segment->get_applications()) {
97 if (app->config_object().implementation() == child)
98 out.push_back(p_list);
99 else if (const auto resource_set = app->cast<dunedaq::coredal::ResourceSet>())
100 make_parents_list(child, resource_set, p_list, out, cd_fuse);
101 }
102 }
103
104 // remove the segment from the path
105
106 p_list.pop_back();
107}

132{
133 const ConfigObjectImpl * obj_impl = config_object().implementation();
134
135 const bool is_segment = castable(dunedaq::coredal::Segment::s_class_name);
136
137 try {
138 dunedaq::coredal::TestCircularDependency cd_fuse("component parents", &session);
139

142 cd_fuse);
143
144
145 if (parents.empty()) {
146 TLOG_DEBUG(1) << "cannot find segment/resource path(s) between Component " << this << " and session " << &session << " objects (check this object is linked with the session as a segment or a resource)" ;
147 }
148 }
149 catch (ers::Issue & ex) {
150 ers::error(CannotGetParents(ERS_HERE, full_name(), ex));
151 }

186nlohmann::json get_json_config(conffwk::Configuration& confdb,
187 const std::string& class_name,
188 const std::string& uid,
189 bool direct_only) {
190 using nlohmann::json;
191 using namespace conffwk;
192 TLOG_DBG(9) << "Getting attributes for " << uid << " of class " << class_name;
193 json attributes;
194 auto class_info = confdb.get_class_info(class_name);
195 ConfigObject obj;
196 confdb.get(class_name, uid, obj);
197 for (auto attr : class_info.p_attributes) {
198 if (attr.p_type == type_t::u8_type) {
199 add_json_value<uint8_t>(obj, attr.p_name, attr.p_is_multi_value, attributes);
200 }
201 else if (attr.p_type == type_t::u16_type) {
202 add_json_value<uint16_t>(obj, attr.p_name, attr.p_is_multi_value, attributes);
203 }
204 else if (attr.p_type == type_t::u32_type) {
205 add_json_value<uint32_t>(obj, attr.p_name, attr.p_is_multi_value, attributes);
206 }
207 else if (attr.p_type == type_t::u64_type) {
208 add_json_value<uint64_t>(obj, attr.p_name, attr.p_is_multi_value, attributes);
209 }
210 else if (attr.p_type == type_t::s8_type) {
211 add_json_value<int8_t>(obj, attr.p_name, attr.p_is_multi_value, attributes);
212 }
213 else if (attr.p_type == type_t::s16_type) {
214 add_json_value<int16_t>(obj, attr.p_name, attr.p_is_multi_value, attributes);
215 }
216 else if (attr.p_type == type_t::s32_type ||
217 attr.p_type == type_t::s16_type) {
218 add_json_value<int32_t>(obj, attr.p_name, attr.p_is_multi_value, attributes);
219 }
220 else if (attr.p_type == type_t::s64_type) {
221 add_json_value<int64_t>(obj, attr.p_name, attr.p_is_multi_value, attributes);
222 }
223 else if (attr.p_type == type_t::float_type) {
224 add_json_value<float>(obj, attr.p_name, attr.p_is_multi_value, attributes);
225 }
226 else if (attr.p_type == type_t::double_type) {
227 add_json_value<double>(obj, attr.p_name, attr.p_is_multi_value, attributes);
228 }
229 else if (attr.p_type == type_t::bool_type) {
230 add_json_value<bool>(obj, attr.p_name, attr.p_is_multi_value, attributes);
231 }
232 else if ((attr.p_type == type_t::string_type) ||
233 (attr.p_type == type_t::enum_type) ||
234 (attr.p_type == type_t::date_type) ||
235 (attr.p_type == type_t::time_type)) {
236 add_json_value<std::string>(obj, attr.p_name, attr.p_is_multi_value, attributes);
237 }
238 }
239 if (!direct_only) {
240 TLOG_DBG(9) << "Processing relationships";
241 for (auto iter: class_info.p_relationships) {
242 std::string rel_name = iter.p_name;
243 if (iter.p_cardinality == cardinality_t::zero_or_one ||
244 iter.p_cardinality == cardinality_t::only_one) {
245 ConfigObject rel_obj;
246 obj.get(rel_name, rel_obj);
247 if (!rel_obj.is_null()) {
248 TLOG_DBG(9) << "Getting attibute of relationship " << rel_name;
249 attributes[rel_name] = get_json_config(confdb, rel_obj.class_name(),
250 rel_obj.UID(),
251 direct_only);
252 }
253 else {
254 TLOG_DBG(9) << "Relationship " << rel_name << " not set";
255 }
256 }
257 else {
258 TLOG_DBG(9) << "Relationship " << rel_name << " is multi value. "
259 << "Getting attibutes for relationship.";
260 std::vector<ConfigObject> rel_vec;
261 obj.get(rel_name, rel_vec);
262 std::vector<json> configs;
263 for (auto rel_obj : rel_vec) {
264 TLOG_DBG(9) << "Getting attibute of relationship " << rel_obj.UID();
265 auto rel_conf = get_json_config(confdb, rel_obj.class_name(), rel_obj.UID(),
266 direct_only);
267 configs.push_back(rel_conf);
268 }
269 attributes[rel_name] = configs;
270 }
271 }
272 }
273 json json_config;
274 json_config[uid] = attributes;
275 return json_config;
276}

308 + ":"
309 + std::to_string(control_service->get_port());

311 std::vector<std::string> ret = {};
312 ret.push_back(configuration_uri);
313 ret.push_back(control_uri);
314 ret.push_back(UID());
315 ret.push_back(session->UID());
316 return ret;
