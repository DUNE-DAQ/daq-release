79 const dunedaq::coredal::Session* session_ptr = const_cast<Configuration&>(db).get<dunedaq::coredal::Session>(session_id);
80
81 std::list<std::vector<const dunedaq::coredal::Component*>> parents;
82 std::vector<std::vector<ObjectLocator>> parent_ids;
83
84 component_ptr->get_parents(*session_ptr, parents);
85
86 for (const auto& parent : parents) {
87 std::vector<ObjectLocator> parents_components;
88 for (const auto& ancestor_component_ptr : parent) {
89 parents_components.emplace_back(
90 ObjectLocator(ancestor_component_ptr->UID(),
91 ancestor_component_ptr->class_name()) );
92 }
93 parent_ids.emplace_back(parents_components);
94 }
