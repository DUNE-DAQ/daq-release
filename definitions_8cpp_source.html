30std::map<std::string, FixedLengthCommandType> swap_commands_map(const std::map<FixedLengthCommandType, std::string>& command_map)
31{
32 std::map<std::string, FixedLengthCommandType> swapped_map;
33 for (auto& cmd: command_map) {
34 swapped_map.emplace( std::pair<std::string,FixedLengthCommandType>(cmd.second, cmd.first) );
35 }
36 return swapped_map;
37}
