52void
53TLUIONode::reset(const std::string& clock_config_file) const
54{
55 // Soft reset
56 write_soft_reset_register();
57
58 millisleep(1000);
59
60 // Reset PLL and I2C
61 getNode("csr.ctrl.pll_rst").write(0x1);
62 getNode("csr.ctrl.pll_rst").write(0x0);
63
64 getNode("csr.ctrl.rst_i2c").write(0x1);
65 getNode("csr.ctrl.rst_i2c").write(0x0);
66
67 getClient().dispatch();
68
69 // enclustra i2c switch stuff
70 try {
71 getNode<I2CMasterNode>(m_uid_i2c_bus).get_slave("AX3_Switch").write_i2c(0x01, 0x7f);
72 } catch (const std::exception& e) {
73 ers::warning(EnclustraSwitchFailure(ERS_HERE, e));
74 }
75
76 // Find the right pll config file
77 std::string clock_config_path = get_full_clock_config_file_path(clock_config_file);
78 TLOG() << "PLL configuration file : " << clock_config_path;
79
80 // Upload config file to PLL
81 configure_pll(clock_config_path);
82
83 // Tweak the PLL swing
84 auto si_chip = get_pll();
85 si_chip->write_i2cArray(0x113, { 0x9, 0x33 });
86
87 // Reset mmcm
88 getNode("csr.ctrl.rst").write(0x1);
89 getNode("csr.ctrl.rst").write(0x0);
90 getClient().dispatch();
91
92 // configure tlu io expanders
93 auto ic_6 = get_i2c_device<I2CExpanderSlave>(m_uid_i2c_bus, "Expander1");
94 auto ic_7 = get_i2c_device<I2CExpanderSlave>(m_uid_i2c_bus, "Expander2");
95
96 // Bank 0
97 ic_6->set_inversion(0, 0x00);
98 ic_6->set_io(0, 0x00);
99 ic_6->set_outputs(0, 0x00);
100
101 // Bank 1
102 ic_6->set_inversion(1, 0x00);
103 ic_6->set_io(1, 0x00);
104 ic_6->set_outputs(1, 0x88);
105
106 // Bank 0
107 ic_7->set_inversion(0, 0x00);
108 ic_7->set_io(0, 0x00);
109 ic_7->set_outputs(0, 0xf0);
110
111 // Bank 1
112 ic_7->set_inversion(1, 0x00);
113 ic_7->set_io(1, 0x00);
114 ic_7->set_outputs(1, 0xf0);
115
116 // BI signals are NIM
117 uint32_t bi_signal_threshold = 0x589D; // NOLINT(build/unsigned)
118
119 configure_dac(0, bi_signal_threshold);
120 configure_dac(1, bi_signal_threshold);
121
122 TLOG_DEBUG(0) << "DAC1 and DAC2 set to " << std::hex << bi_signal_threshold;
123
124 TLOG() << "Reset done";
125}
