38PDIEndpointNode::enable(uint32_t address, uint32_t partition) const // NOLINT(build/unsigned)
39{
40 getNode("csr.ctrl.tgrp").write(partition);

53 getClient().dispatch();
54}

80std::string
81PDIEndpointNode::get_status(bool print_out) const
82{
83
84 std::stringstream status;
85
86 std::vector<std::pair<std::string, std::string>> ept_summary;
87
88 auto ept_timestamp = getNode("tstamp").readBlock(2);
89 auto ept_event_counter = getNode("evtctr").read();
90 auto ept_buffer_count = getNode("buf.count").read();
91 auto ept_control = read_sub_nodes(getNode("csr.ctrl"), false);
92 auto ept_state = read_sub_nodes(getNode("csr.stat"), false);
93 auto ept_counters = getNode("ctrs").readBlock(PDIFLCmdGeneratorNode::number_of_fl_cmds);
94 getClient().dispatch();
95
96 ept_summary.push_back(std::make_pair("State", get_endpoint_state_map().at(ept_state.find("ep_stat")->second.value())));
97 ept_summary.push_back(std::make_pair("Partition", std::to_string(ept_control.find("tgrp")->second.value())));
98 ept_summary.push_back(std::make_pair("Address", std::to_string(ept_control.find("addr")->second.value())));
99
100 auto ept_clock_frequency = read_clock_frequency();
101
102 if (abs((ept_clock_frequency*1e6)-62.5e6) < 10e3) {
103 ept_summary.push_back(std::make_pair("Timestamp", format_timestamp(ept_timestamp,62500000)));
104 } else if (abs((ept_clock_frequency*1e6)-50e6) < 10e3) {
105 ept_summary.push_back(std::make_pair("Timestamp", format_timestamp(ept_timestamp,50000000)));
106 } else {
107 ept_summary.push_back(std::make_pair("Timestamp", format_timestamp(ept_timestamp,ept_clock_frequency)));
108 }
109
110 ept_summary.push_back(std::make_pair("Timestamp (hex)", format_reg_value(tstamp2int(ept_timestamp))));
111 ept_summary.push_back(std::make_pair("EventCounter", std::to_string(ept_event_counter.value())));
112 std::string buffer_status_string = !ept_state.find("buf_err")->second.value() ? "OK" : "Error";
113 ept_summary.push_back(std::make_pair("Buffer status", buffer_status_string));
114 ept_summary.push_back(std::make_pair("Buffer occupancy", std::to_string(ept_buffer_count.value())));
115
116 std::vector<std::pair<std::string, std::string>> ept_command_counters;
117
118 for (auto& cmd : PDIFLCmdGeneratorNode::get_command_map()) {
119 ept_command_counters.push_back(std::make_pair(cmd.second, std::to_string(ept_counters[cmd.first])));
120 }
121
122 status << format_reg_table(ept_summary, "Endpoint summary", { "", "" }) << std::endl;
123 status << "Endpoint frequency: " << ept_clock_frequency << " MHz" << std::endl;
124 status << format_reg_table(ept_state, "Endpoint state") << std::endl;
125 status << format_reg_table(ept_command_counters, "Endpoint counters", { "Command", "Counter" });
126
127 if (print_out)
128 TLOG() << status.str();
129 return status.str();
130}

154uhal::ValVector<uint32_t> // NOLINT(build/unsigned)
155PDIEndpointNode::read_data_buffer(bool read_all) const
156{
157
158 auto buffer_count = getNode("buf.count").read();
159 getClient().dispatch();
160
161 TLOG_DEBUG(0) << "Words available in readout buffer: " << format_reg_value(buffer_count);
162
163 uint32_t events_to_read = buffer_count.value() / PartitionNode::kWordsPerEvent; // NOLINT(build/unsigned)
164
165 TLOG_DEBUG(0) << "Events available in readout buffer: " << format_reg_value(events_to_read);
166
167 uint32_t words_to_read = read_all ? buffer_count.value() : events_to_read * PartitionNode::kWordsPerEvent; // NOLINT(build/unsigned)
168
169 TLOG_DEBUG(0) << "Words to be read out in readout buffer: " << format_reg_value(words_to_read);
170
171 if (!words_to_read) {
172 TLOG() << "No words to be read out.";
173 }
174
175 auto buffer_data = getNode("buf.data").readBlock(words_to_read);
176 getClient().dispatch();
177
178 return buffer_data;
179}

183std::string
184PDIEndpointNode::get_data_buffer_table(bool read_all, bool print_out) const
185{
186
187 std::stringstream table;
188 auto buffer_data = read_data_buffer(read_all);
189
190 std::vector<std::pair<std::string, uint32_t>> buffer_table; // NOLINT(build/unsigned)
191
192 uint32_t i = 0; // NOLINT(build/unsigned)
193 for (auto it = buffer_data.begin(); it != buffer_data.end(); ++it, ++i) {
194 std::stringstream index_stream;
195 index_stream << std::setfill('0') << std::setw(4) << i;
196 buffer_table.push_back(std::make_pair(index_stream.str(), *it));
197 }
198 table << format_reg_table(buffer_table, "Endpoint buffer", { "Word", "Data" });
199
200 if (print_out)
201 TLOG() << table.str();
202 return table.str();
