62void
63TPProcessor::conf(const appdal::ReadoutModule* conf)
64{
65 for (auto output : conf->get_outputs()) {
66 try {
67 if (output->get_data_type() == "TriggerActivity") {
68 m_ta_sink = get_iom_sender<triggeralgs::TriggerActivity>(output->UID());
69 }
70 } catch (const ers::Issue& excpt) {
71 ers::error(readoutlibs::ResourceQueueError(ERS_HERE, "ta", "DefaultRequestHandlerModel", excpt));
72 }
73 }
74
75 m_sourceid.id = conf->get_source_id();
76 m_sourceid.subsystem = TriggerPrimitiveTypeAdapter::subsystem;
77
78 std::vector<const appdal::TAAlgorithm*> ta_algorithms;
79 auto dp = conf->get_module_configuration()->get_data_processor();
80 auto proc_conf = dp->cast<appdal::TPDataProcessor>();
81 if (proc_conf != nullptr && proc_conf->get_mask_processing() == false) {
82 ta_algorithms = proc_conf->get_algorithms();
83 }
84
85 for (auto algo : ta_algorithms) {
86 TLOG() << "Selected TA algorithm: " << algo->UID() << " from class " << algo->class_name();
87 std::shared_ptr<triggeralgs::TriggerActivityMaker> maker = make_ta_maker(algo->class_name());
88 nlohmann::json algo_json = algo->to_json(true);
89
90 TLOG() << "Algo config:\n" << algo_json.dump();
91
92 maker->configure(algo_json[algo->UID()]);
93 inherited::add_postprocess_task(std::bind(&TPProcessor::find_ta, this, std::placeholders::_1, maker));
94 m_tams.push_back(maker);
95 }
96 inherited::conf(conf);
97}

111void
112TPProcessor::find_ta(const TriggerPrimitiveTypeAdapter* tp, std::shared_ptr<triggeralgs::TriggerActivityMaker> taa)
113{
114
115 std::vector<triggeralgs::TriggerActivity> tas;
116 taa->operator()(tp->tp, tas);
117
118 while (tas.size()) {
119 if (!m_ta_sink->try_send(std::move(tas.back()), iomanager::Sender::s_no_block)) {
120 ers::warning(TADropped(ERS_HERE, tp->tp.time_start, m_sourceid.id));
121 m_tas_dropped++;
122 }
123 m_new_tas++;
124 tas.pop_back();
125 }
126 return;
127}
