136uint32_t // NOLINT(build/unsigned)
137PDIMasterNode::measure_endpoint_rtt(uint32_t address, bool control_sfp) const // NOLINT(build/unsigned)
138{
139
140 auto vl_cmd_node = getNode<VLCmdGeneratorNode>("acmd");
141 auto global = getNode<PDIMasterGlobalNode>("global");
142 auto echo = getNode<EchoMonitorNode>("echo");
143
144 if (control_sfp)
145 {
146 // Switch off all TX SFPs
147 vl_cmd_node.switch_endpoint_sfp(0x0, false);
148
149 // Turn on the current target
150 vl_cmd_node.switch_endpoint_sfp(address, true);
151
152 millisleep(200);
153 }
154
155 try
156 {
157 global.enable_upstream_endpoint();
158 }
159 catch (const timing::EndpointNotReady& e)
160 {
161 if (control_sfp) {
162 vl_cmd_node.switch_endpoint_sfp(address, false);
163 }
164 throw e;
165 }
166 uint32_t endpoint_rtt = echo.send_echo_and_measure_delay(); // NOLINT(build/unsigned)
167
168 if (control_sfp)
169 vl_cmd_node.switch_endpoint_sfp(address, false);
170
171 return endpoint_rtt;
172}

177PDIMasterNode::apply_endpoint_delay(uint32_t address, // NOLINT(build/unsigned)
178 uint32_t coarse_delay, // NOLINT(build/unsigned)
179 uint32_t fine_delay, // NOLINT(build/unsigned)
180 uint32_t phase_delay, // NOLINT(build/unsigned)

183{
184
185 auto vl_cmd_node = getNode<VLCmdGeneratorNode>("acmd");
186 auto global = getNode<PDIMasterGlobalNode>("global");
187 auto echo = getNode<EchoMonitorNode>("echo");
188
189 if (measure_rtt) {
190 uint64_t endpoint_rtt = measure_endpoint_rtt(address, control_sfp); // NOLINT(build/unsigned)
191 TLOG() << "Pre delay adjustment RTT: " << format_reg_value(endpoint_rtt,10);

199 }

267timingfirmware::EndpointCheckResult
268PDIMasterNode::scan_endpoint(uint16_t endpoint_address, bool control_sfp) const
269{
270 timingfirmware::EndpointCheckResult result;
271 auto global = getNode<PDIMasterGlobalNode>("global");
272 auto echo = getNode<EchoMonitorNode>("echo");
273
274 timingfirmware::EndpointCheckResult endpoint_result;
275 endpoint_result.address = endpoint_address;
276
277 if (control_sfp)
278 {
279 switch_endpoint_sfp(endpoint_address, true);
280
281 millisleep(100);
282 }
283
284 try
285 {
286 global.enable_upstream_endpoint();
287 }
288 catch (const timing::EndpointNotReady& e)
289 {
290 switch_endpoint_sfp(endpoint_address, false);
291
292 TLOG_DEBUG(5) << "Endpoint at address " << endpoint_address << " looks dead.";
293
294 return endpoint_result;
295 }
296
297 endpoint_result.alive = true;
298 endpoint_result.round_trip_time = echo.send_echo_and_measure_delay();
299 TLOG_DEBUG(5) << "Endpoint at address " << endpoint_address << " alive. RTT: " << endpoint_result.round_trip_time;
300
301 if (control_sfp)
302 {
303 switch_endpoint_sfp(endpoint_address, false);
304 }
305
306 return endpoint_result;
307}
