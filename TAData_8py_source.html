110 def load_frag(self, frag_path, index=None) -> (list, list):
111 """
112 Load a fragment from a given fragment path.
113
114 Returns two lists: TA data and TP data.
115 """
116 frag = self._h5_file.get_frag(frag_path)
117 frag_data_size = frag.get_data_size()
118 if frag_data_size == 0:
119 if not self._quiet:
120 print(self._FAIL_TEXT_COLOR + self._BOLD_TEXT + "WARNING: Empty fragment. Returning empty array." + self._END_TEXT_COLOR_END_TEXT_COLOR)
121 print(self._WARNING_TEXT_COLOR + self._BOLD_TEXT + f"INFO: Fragment Path: {frag_path}" + self._END_TEXT_COLOR_END_TEXT_COLOR)
122 if index != None:
123 self._nonempty_frags_mask[index] = False
124 return list(), list()
125 if index == None:
126 index = self._frag_paths.index(frag_path)
127
128 # Unknown number of TAs or TPs per TA.
129 frag_ta_data = []
130 frag_tp_data = []
131
132 ta_idx = 0 # Only used to output.
133 byte_idx = 0 # Variable TA sizing, must do while loop.
134 while (byte_idx < frag_data_size):
135 if (not self._quiet):
136 print(f"Fragment Index: {ta_idx}.")
137 ta_idx += 1
138 print(f"Byte Index / Frag Size: {byte_idx} / {frag_data_size}")
139
140 ta_datum = trgdataformats.TriggerActivity(frag.get_data(byte_idx))
141 np_ta_datum = np.array([(
142 ta_datum.data.adc_integral,
143 ta_datum.data.adc_peak,
144 np.uint8(ta_datum.data.algorithm),
145 ta_datum.data.channel_end,
146 ta_datum.data.channel_peak,
147 ta_datum.data.channel_start,
148 np.uint16(ta_datum.data.detid),
149 ta_datum.n_inputs(),
150 ta_datum.data.time_activity,
151 ta_datum.data.time_end,
152 ta_datum.data.time_peak,
153 ta_datum.data.time_start,
154 np.uint8(ta_datum.data.type))],
155 dtype=self.ta_dt)
156 frag_ta_data.append(np_ta_datum)
157 byte_idx += ta_datum.sizeof()
158 if (not self._quiet):
159 print(f"Upcoming byte: {byte_idx}")
160
161
162 num_tps = ta_datum.n_inputs()
163 np_tp_data = np.zeros((num_tps,), dtype=self.tp_dt)
164 for tp_idx, tp in enumerate(ta_datum):
165 np_tp_data[tp_idx] = np.array([(

168 tp.algorithm,
169 tp.channel,
170 tp.detid,
171 tp.flag,
172 tp.time_over_threshold,
173 tp.time_peak,
174 tp.time_start,
175 tp.type,
176 tp.version)],
177 dtype=self.tp_dt)
178 frag_tp_data.append(np_tp_data) # Jagged array
179 self.tp_data.append(frag_tp_data)
180 self.ta_data.append(frag_ta_data)
