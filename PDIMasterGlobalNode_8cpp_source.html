91{
92 getNode("csr.ctrl.ep_en").write(0x0);
93 getClient().dispatch();
94 getNode("csr.ctrl.ep_en").write(0x1);
95 getClient().dispatch();
96
97 TLOG() << "Upstream endpoint reset, waiting for lock";
98
99 auto start = std::chrono::high_resolution_clock::now();
100
101 std::chrono::milliseconds ms_since_start(0);
102
103 uhal::ValWord<uint32_t> ept_state; // NOLINT(build/unsigned)
104 uhal::ValWord<uint32_t> ept_ready; // NOLINT(build/unsigned)
105
106 // Wait for the endpoint to be happy
107 while (ms_since_start.count() < timeout) {
108 auto now = std::chrono::high_resolution_clock::now();
109 ms_since_start = std::chrono::duration_cast<std::chrono::milliseconds>(now - start);
110
111 millisleep(50);
112
113 ept_state = getNode("csr.stat.ep_stat").read();
114 ept_ready = getNode("csr.stat.ep_rdy").read();
115
116 auto ept_edge = getNode("csr.stat.ep_edge").read();
117 auto ept_fdel = getNode("csr.stat.ep_fdel").read();
118
119 getClient().dispatch();
120
121 TLOG_DEBUG(1) << "ept state: 0x" << std::hex << ept_state.value() << " ept ready: 0x" << ept_ready.value() << " ept edge: 0x" << ept_edge.value() << " ept fdel: 0x" << ept_fdel.value();
122
123 if (ept_ready.value() && ept_state.value() == 0x8) {
124 TLOG_DEBUG(1) << "Endpoint locked: state= " << format_reg_value(ept_state);
125 return;
126 }
127 }
128
129 if (!ept_ready.value() || ept_state.value() != 0x8) {
130 throw EndpointNotReady(ERS_HERE, "Master upstream", ept_state.value());
131 } else {
132 TLOG_DEBUG(1) << "Endpoint locked: state= " << format_reg_value(ept_state);
133 }
134}
